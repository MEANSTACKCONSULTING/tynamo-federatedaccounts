001package org.tynamo.security.federatedaccounts.pac4j.components;
002
003import org.apache.tapestry5.Asset;
004import org.apache.tapestry5.BindingConstants;
005import org.apache.tapestry5.ComponentResources;
006import org.apache.tapestry5.annotations.Parameter;
007import org.apache.tapestry5.annotations.Path;
008import org.apache.tapestry5.annotations.Property;
009import org.apache.tapestry5.ioc.annotations.Inject;
010import org.apache.tapestry5.services.AssetSource;
011import org.apache.tapestry5.services.PageRenderLinkSource;
012import org.pac4j.core.exception.CommunicationException;
013import org.tynamo.security.federatedaccounts.FederatedAccount.FederatedAccountType;
014import org.tynamo.security.federatedaccounts.base.AbstractOauthSignIn;
015import org.tynamo.security.federatedaccounts.pac4j.pages.Pac4jOauth;
016import org.tynamo.security.federatedaccounts.pac4j.services.Pac4jOauthClientLocator;
017import org.tynamo.security.federatedaccounts.util.WindowMode;
018
019public class Pac4jOauthSignIn extends AbstractOauthSignIn {
020        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "800")
021        @Property
022        private Integer width;
023
024        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "400")
025        @Property
026        private Integer height;
027
028        @Parameter(value = "blank", required = false, defaultPrefix = "literal")
029        private WindowMode windowMode;
030
031        @Inject
032        private AssetSource assetSource;
033
034        @Inject
035        private ComponentResources componentResources;
036
037        @Inject
038        private PageRenderLinkSource pageRenderLinkSource;
039
040        @Parameter(required = true, defaultPrefix = BindingConstants.LITERAL)
041        @Property(write = false)
042        private String provider;
043
044        public Asset getSignInImage() {
045                return assetSource.getAsset(componentResources.getBaseResource(), provider + ".png", null);
046        }
047
048        public boolean isWindowMode(String mode) {
049                if (mode == null) throw new IllegalArgumentException("Window mode argument cannot be null");
050                return windowMode.equals(WindowMode.valueOf(mode));
051        }
052
053        @Inject
054        private PageRenderLinkSource linkSource;
055
056        @Override
057        protected Class<?> getOauthPageClass() {
058                return Pac4jOauth.class;
059        }
060
061        public String getOauthAuthorizationLink() throws CommunicationException {
062                return getOauthRedirectLink(windowMode, provider, "request_token", getReturnPageUri());
063        }
064
065        public String getProviderPrefix() {
066                return FederatedAccountType.pac4j_ + provider;
067        }
068
069        @Inject
070        @Path("classpath:org/tynamo/security/federatedaccounts/pac4j/components/supportedclients.png")
071        private Asset clientSprites;
072
073        public String getButtonSpriteStyle() {
074                // given that displaying the right sprite requires specific css, let's just create the whole style here
075                // we could later support supplying custom style and auth method as well but might be easier just to create a separate component for
076                // that
077
078                Pac4jOauthClientLocator.SupportedClient client = null;
079                try {
080                        client = Pac4jOauthClientLocator.SupportedClient.valueOf(provider);
081                } catch (IllegalArgumentException e) {
082                        logger.error("Provider type " + provider + " isn't supported by current Pac4jOauth component");
083                        return "";
084                }
085
086                StringBuilder sb = new StringBuilder();
087                sb.append("width:150px;height:50px;background:transparent url('");
088                sb.append(clientSprites);
089                sb.append("') no-repeat 0 ");
090                sb.append(-50 * client.ordinal());
091                sb.append("px;");
092                return sb.toString();
093        }
094}
