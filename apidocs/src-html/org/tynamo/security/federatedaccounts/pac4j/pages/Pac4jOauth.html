001package org.tynamo.security.federatedaccounts.pac4j.pages;
002
003import java.net.URL;
004
005import javax.servlet.http.HttpServletRequest;
006import javax.servlet.http.HttpServletResponse;
007
008import org.apache.shiro.SecurityUtils;
009import org.apache.shiro.authc.AuthenticationException;
010import org.apache.tapestry5.EventContext;
011import org.apache.tapestry5.alerts.AlertManager;
012import org.apache.tapestry5.ioc.annotations.Inject;
013import org.apache.tapestry5.ioc.annotations.Symbol;
014import org.apache.tapestry5.services.Cookies;
015import org.apache.tapestry5.services.Request;
016import org.pac4j.core.context.J2EContext;
017import org.pac4j.core.profile.UserProfile;
018import org.pac4j.oauth.client.BaseOAuthClient;
019import org.pac4j.oauth.client.DropBoxClient;
020import org.pac4j.oauth.credentials.OAuthCredentials;
021import org.slf4j.Logger;
022import org.tynamo.security.federatedaccounts.FederatedAccount.FederatedAccountType;
023import org.tynamo.security.federatedaccounts.FederatedAccountSymbols;
024import org.tynamo.security.federatedaccounts.base.AbstractOauthPage;
025import org.tynamo.security.federatedaccounts.pac4j.Pac4jAuthenticationToken;
026import org.tynamo.security.federatedaccounts.pac4j.services.Pac4jOauthClientLocator;
027
028public class Pac4jOauth extends AbstractOauthPage {
029        @Inject
030        @Symbol(FederatedAccountSymbols.HTTPCLIENT_ON_GAE)
031        private boolean httpClientOnGae;
032
033        @Inject
034        private Logger logger;
035
036        @Inject
037        private Request request;
038
039        @Inject
040        private HttpServletRequest httpRequest;
041
042        @Inject
043        private HttpServletResponse httpResponse;
044
045        @Inject
046        private AlertManager alertManager;
047
048        // private String returnUri;
049
050        @Inject
051        private Cookies cookies;
052
053        @Inject
054        private Pac4jOauthClientLocator oauthClientLocator;
055
056        private String clientName;
057
058        @Override
059        protected Object onOauthActivate(EventContext eventContext) throws Exception {
060                clientName = eventContext.get(String.class, 1);
061                BaseOAuthClient<?> client = oauthClientLocator.getClient(clientName);
062                client.setReadTimeout(20000);
063                client.setConnectTimeout(20000);
064                if (eventContext.getCount() > 3) {
065                        String action = eventContext.get(String.class, 2);
066                        // pass along this redirectUrl
067                        setReturnUri(eventContext.get(String.class, 3));
068                        client.setCallbackUrl(getOauthRedirectLink(getWindowMode(), clientName));
069                        if ("request_token".equals(action)) {
070                                String providerOauthUrl = client.getRedirectionUrl(new J2EContext(httpRequest, httpResponse), true);
071                                // fix an issue with pac4j dropboxClient and "direct redirection"
072                                if (client instanceof DropBoxClient)
073                                        if (!providerOauthUrl.contains("&oauth_callback"))
074                                                providerOauthUrl += "&oauth_callback=" + client.getCallbackUrl();
075                                // Because Google requires the callback uri to match *exactly* (including query parameters) with the configuration,
076                                // use a cookie to store the destination uri after successful Oauth authentication
077                                cookies.writeCookieValue("oauth_returnpage", getReturnUri(), -1);
078                                return new URL(providerOauthUrl);
079                        }
080                }
081                // FIXME should we catch and handle the event context errors? Probably
082
083                // capture the redirect url again
084                String returnPageUri = cookies.readCookieValue("oauth_returnpage");
085                if (returnPageUri != null) {
086                        setReturnUri(returnPageUri);
087                        cookies.removeCookieValue("oauth_returnpage");
088                }
089
090                // we'll get a technical exception if callback url is not set even though it shouldn't be needed anymore
091                client.setCallbackUrl(getOauthRedirectLink(getWindowMode(), clientName));
092
093                OAuthCredentials credentials = client.getCredentials(new J2EContext(httpRequest, httpResponse));
094
095                UserProfile userProfile = client.getUserProfile(credentials);
096                Pac4jAuthenticationToken accessToken = new Pac4jAuthenticationToken(userProfile);
097
098                // TODO just use the default rememberMe for now. We could later add support for providing rememberMe in context
099                accessToken.setRememberMe(isRememberMe());
100
101                try {
102                        SecurityUtils.getSubject().login(accessToken);
103                        alertManager.success("User successfully authenticated");
104                        setOauthAuthenticated(true);
105                } catch (AuthenticationException e) {
106                        logger.error(
107                                "Using access token " + accessToken
108                                        + String.format("\nCould not sign in a %s federated user because of: ", credentials.getClientName()), e);
109                        // FIXME Deal with other account exception types like expired and
110                        // locked
111                        alertManager.error(String.format("A %s federated user cannot be signed in, report this to support.\n %s",
112                                credentials.getClientName(), e.getMessage()));
113                }
114                client = null;
115                return null;
116        }
117
118        public String getProviderPrefix() {
119                return FederatedAccountType.pac4j_ + clientName;
120        }
121}
