001package org.tynamo.security.federatedaccounts.pac4j;
002
003import org.apache.shiro.realm.AuthenticatingRealm;
004import org.apache.shiro.realm.Realm;
005import org.apache.tapestry5.ioc.Configuration;
006import org.apache.tapestry5.ioc.MappedConfiguration;
007import org.apache.tapestry5.ioc.ServiceBinder;
008import org.apache.tapestry5.ioc.annotations.Contribute;
009import org.apache.tapestry5.ioc.annotations.InjectService;
010import org.apache.tapestry5.services.LibraryMapping;
011import org.tynamo.common.ModuleProperties;
012import org.tynamo.security.federatedaccounts.FederatedAccount.FederatedAccountType;
013import org.tynamo.security.federatedaccounts.pac4j.services.Pac4jFederatedRealm;
014import org.tynamo.security.federatedaccounts.pac4j.services.Pac4jOauthClientLocator;
015import org.tynamo.security.federatedaccounts.services.FederatedAccountsModule;
016import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentBlockSource;
017import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentContribution;
018
019public class Pac4jFederatedAccountsModule {
020        private static final String PATH_PREFIX = "federated-pac4j";
021        private static String version = ModuleProperties.getVersion(FederatedAccountsModule.class);
022
023        public static void bind(ServiceBinder binder) {
024                binder.bind(AuthenticatingRealm.class, Pac4jFederatedRealm.class).withId(Pac4jFederatedRealm.class.getSimpleName());
025                binder.bind(Pac4jOauthClientLocator.class);
026        }
027
028        public static void contributeFactoryDefaults(MappedConfiguration<String, String> configuration) {
029                configuration.add(Pac4jFederatedRealm.DROPBOX_CLIENTID, "");
030                configuration.add(Pac4jFederatedRealm.DROPBOX_CLIENTSECRET, "");
031                configuration.add(Pac4jFederatedRealm.FACEBOOK_CLIENTID, "");
032                configuration.add(Pac4jFederatedRealm.FACEBOOK_CLIENTSECRET, "");
033                configuration.add(Pac4jFederatedRealm.GITHUB_CLIENTID, "");
034                configuration.add(Pac4jFederatedRealm.GITHUB_CLIENTSECRET, "");
035                configuration.add(Pac4jFederatedRealm.GOOGLE_CLIENTID, "");
036                configuration.add(Pac4jFederatedRealm.GOOGLE_CLIENTSECRET, "");
037                configuration.add(Pac4jFederatedRealm.LINKEDIN_CLIENTID, "");
038                configuration.add(Pac4jFederatedRealm.LINKEDIN_CLIENTSECRET, "");
039                configuration.add(Pac4jFederatedRealm.TWITTER_CLIENTID, "");
040                configuration.add(Pac4jFederatedRealm.TWITTER_CLIENTSECRET, "");
041                configuration.add(Pac4jFederatedRealm.WINDOWSLIVE_CLIENTID, "");
042                configuration.add(Pac4jFederatedRealm.WINDOWSLIVE_CLIENTSECRET, "");
043                configuration.add(Pac4jFederatedRealm.WORDPRESS_CLIENTID, "");
044                configuration.add(Pac4jFederatedRealm.WORDPRESS_CLIENTSECRET, "");
045                configuration.add(Pac4jFederatedRealm.YAHOO_CLIENTID, "");
046                configuration.add(Pac4jFederatedRealm.YAHOO_CLIENTSECRET, "");
047        }
048
049        public static void contributeComponentClassResolver(Configuration<LibraryMapping> configuration) {
050                configuration.add(new LibraryMapping(FederatedAccountsModule.PATH_PREFIX,
051                        "org.tynamo.security.federatedaccounts.pac4j"));
052        }
053
054        public static void contributeClasspathAssetAliasManager(MappedConfiguration<String, String> configuration) {
055                configuration.add(PATH_PREFIX + "-" + version, "org/tynamo/security/federatedaccounts/pac4j");
056        }
057
058        public static void contributeWebSecurityManager(Configuration<Realm> configuration,
059                @InjectService("Pac4jFederatedRealm") AuthenticatingRealm pac4jRealm) {
060                configuration.add(pac4jRealm);
061        }
062
063        @Contribute(FederatedSignInComponentBlockSource.class)
064        public static void addSignInComponentBlocks(Configuration<FederatedSignInComponentContribution> configuration) {
065                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
066                        + Pac4jOauthClientLocator.SupportedClient.facebook, "federated/pac4jSignInComponentBlocks"));
067                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
068                        + Pac4jOauthClientLocator.SupportedClient.dropbox, "federated/pac4jSignInComponentBlocks"));
069                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
070                        + Pac4jOauthClientLocator.SupportedClient.github, "federated/pac4jSignInComponentBlocks"));
071                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
072                        + Pac4jOauthClientLocator.SupportedClient.google2, "federated/pac4jSignInComponentBlocks"));
073                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
074                        + Pac4jOauthClientLocator.SupportedClient.linkedin2, "federated/pac4jSignInComponentBlocks"));
075                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
076                        + Pac4jOauthClientLocator.SupportedClient.twitter, "federated/pac4jSignInComponentBlocks"));
077                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
078                        + Pac4jOauthClientLocator.SupportedClient.windowslive, "federated/pac4jSignInComponentBlocks"));
079                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
080                        + Pac4jOauthClientLocator.SupportedClient.wordpress, "federated/pac4jSignInComponentBlocks"));
081                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.pac4j_.name()
082                        + Pac4jOauthClientLocator.SupportedClient.yahoo, "federated/pac4jSignInComponentBlocks"));
083        }
084
085}
