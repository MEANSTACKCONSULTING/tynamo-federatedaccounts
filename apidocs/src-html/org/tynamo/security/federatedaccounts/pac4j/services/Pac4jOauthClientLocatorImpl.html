001package org.tynamo.security.federatedaccounts.pac4j.services;
002
003import org.apache.tapestry5.ioc.annotations.Symbol;
004import org.pac4j.oauth.client.BaseOAuthClient;
005import org.pac4j.oauth.client.DropBoxClient;
006import org.pac4j.oauth.client.FacebookClient;
007import org.pac4j.oauth.client.GitHubClient;
008import org.pac4j.oauth.client.Google2Client;
009import org.pac4j.oauth.client.LinkedIn2Client;
010import org.pac4j.oauth.client.TwitterClient;
011import org.pac4j.oauth.client.WindowsLiveClient;
012import org.pac4j.oauth.client.WordPressClient;
013import org.pac4j.oauth.client.YahooClient;
014
015public class Pac4jOauthClientLocatorImpl implements Pac4jOauthClientLocator {
016
017        private String dropboxClientId;
018        private String dropboxClientSecret;
019        private String facebookClientId;
020        private String facebookClientSecret;
021        private String yahooClientId;
022        private String yahooClientSecret;
023        private String githubClientId;
024        private String githubClientSecret;
025        private String googleClientId;
026        private String googleClientSecret;
027        private String linkedinClientId;
028        private String linkedinClientSecret;
029        private String twitterClientId;
030        private String twitterClientSecret;
031        private String windowsliveClientId;
032        private String windowsliveClientSecret;
033        private String wordpressClientId;
034        private String wordpressClientSecret;
035
036        public Pac4jOauthClientLocatorImpl(@Symbol(Pac4jFederatedRealm.DROPBOX_CLIENTID) String dropboxClientId,
037                @Symbol(Pac4jFederatedRealm.DROPBOX_CLIENTSECRET) String dropboxClientSecret,
038                @Symbol(Pac4jFederatedRealm.FACEBOOK_CLIENTID) String facebookClientId,
039                @Symbol(Pac4jFederatedRealm.FACEBOOK_CLIENTSECRET) String facebookClientSecret,
040                @Symbol(Pac4jFederatedRealm.GITHUB_CLIENTID) String githubClientId,
041                @Symbol(Pac4jFederatedRealm.GITHUB_CLIENTSECRET) String githubClientSecret,
042                @Symbol(Pac4jFederatedRealm.GOOGLE_CLIENTID) String googleClientId,
043                @Symbol(Pac4jFederatedRealm.GOOGLE_CLIENTSECRET) String googleClientSecret,
044                @Symbol(Pac4jFederatedRealm.LINKEDIN_CLIENTID) String linkedinClientId,
045                @Symbol(Pac4jFederatedRealm.LINKEDIN_CLIENTSECRET) String linkedinClientSecret,
046                @Symbol(Pac4jFederatedRealm.TWITTER_CLIENTID) String twitterClientId,
047                @Symbol(Pac4jFederatedRealm.TWITTER_CLIENTSECRET) String twitterClientSecret,
048                @Symbol(Pac4jFederatedRealm.WINDOWSLIVE_CLIENTID) String windowsliveClientId,
049                @Symbol(Pac4jFederatedRealm.WINDOWSLIVE_CLIENTSECRET) String windowsliveClientSecret,
050                @Symbol(Pac4jFederatedRealm.WORDPRESS_CLIENTID) String wordpressClientId,
051                @Symbol(Pac4jFederatedRealm.WORDPRESS_CLIENTSECRET) String wordpressClientSecret,
052                @Symbol(Pac4jFederatedRealm.YAHOO_CLIENTID) String yahooClientId,
053                @Symbol(Pac4jFederatedRealm.YAHOO_CLIENTSECRET) String yahooClientSecret) {
054                this.dropboxClientId = dropboxClientId;
055                this.dropboxClientSecret = dropboxClientSecret;
056                this.facebookClientId = facebookClientId;
057                this.facebookClientSecret = facebookClientSecret;
058                this.githubClientId = githubClientId;
059                this.githubClientSecret = githubClientSecret;
060                this.googleClientId = googleClientId;
061                this.googleClientSecret = googleClientSecret;
062                this.linkedinClientId = linkedinClientId;
063                this.linkedinClientSecret = linkedinClientSecret;
064                this.twitterClientId = twitterClientId;
065                this.twitterClientSecret = twitterClientSecret;
066                this.windowsliveClientId = windowsliveClientId;
067                this.windowsliveClientSecret = windowsliveClientSecret;
068                this.wordpressClientId = wordpressClientId;
069                this.wordpressClientSecret = wordpressClientSecret;
070                this.yahooClientId = yahooClientId;
071                this.yahooClientSecret = yahooClientSecret;
072        }
073
074        @Override
075        public BaseOAuthClient<?> getClient(String clientName) {
076                SupportedClient client = SupportedClient.valueOf(clientName);
077                switch (client) {
078                case dropbox:
079                        return new DropBoxClient(dropboxClientId, dropboxClientSecret);
080                case facebook:
081                        return new FacebookClient(facebookClientId, facebookClientSecret);
082                case github:
083                        return new GitHubClient(githubClientId, githubClientSecret);
084                case google2:
085                        return new Google2Client(googleClientId, googleClientSecret);
086                case linkedin2:
087                        return new LinkedIn2Client(linkedinClientId, linkedinClientSecret);
088                case twitter:
089                        return new TwitterClient(twitterClientId, twitterClientSecret);
090                case windowslive:
091                        return new WindowsLiveClient(windowsliveClientId, windowsliveClientSecret);
092                case wordpress:
093                        return new WordPressClient(wordpressClientId, wordpressClientSecret);
094                case yahoo:
095                        return new YahooClient(yahooClientId, yahooClientSecret);
096                default:
097                        throw new IllegalArgumentException("Client " + clientName + " is not one of the supported clients "
098                                + SupportedClient.values());
099                }
100        }
101}
