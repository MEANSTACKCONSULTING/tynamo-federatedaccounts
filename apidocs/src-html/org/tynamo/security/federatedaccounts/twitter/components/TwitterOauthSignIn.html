001package org.tynamo.security.federatedaccounts.twitter.components;
002
003import org.apache.tapestry5.Asset;
004import org.apache.tapestry5.BindingConstants;
005import org.apache.tapestry5.ComponentResources;
006import org.apache.tapestry5.annotations.Parameter;
007import org.apache.tapestry5.annotations.Property;
008import org.apache.tapestry5.ioc.annotations.Inject;
009import org.apache.tapestry5.services.AssetSource;
010import org.apache.tapestry5.services.PageRenderLinkSource;
011import org.tynamo.security.federatedaccounts.base.AbstractOauthSignIn;
012import org.tynamo.security.federatedaccounts.twitter.pages.TwitterOauth;
013import org.tynamo.security.federatedaccounts.util.WindowMode;
014
015import twitter4j.TwitterException;
016
017public class TwitterOauthSignIn extends AbstractOauthSignIn {
018        // from https://dev.twitter.com/docs/auth/sign-in-with-twitter
019        public enum ButtonStyle {
020                darker("sign-in-with-twitter-d.png"), darker_small("sign-in-with-twitter-d-sm.png"), lighter(
021                        "sign-in-with-twitter-l.png"), ligter_small("sign-in-with-twitter-l-sm.png");
022                private String resourceName;
023
024                ButtonStyle(String resourceName) {
025                        this.resourceName = resourceName;
026                }
027
028                @Override
029                public String toString() {
030                        return resourceName;
031                }
032        }
033
034        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "800")
035        @Property
036        private Integer width;
037
038        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "400")
039        @Property
040        private Integer height;
041
042        @Parameter(value = "darker", required = false, defaultPrefix = "literal")
043        private ButtonStyle buttonStyle;
044
045        @Parameter(value = "blank", required = false, defaultPrefix = "literal")
046        private WindowMode windowMode;
047
048        @Inject
049        private AssetSource assetSource;
050
051        @Inject
052        private ComponentResources componentResources;
053
054        @Inject
055        private PageRenderLinkSource pageRenderLinkSource;
056
057        public Asset getSignInImage() {
058                return assetSource.getAsset(componentResources.getBaseResource(), buttonStyle.toString(), null);
059        }
060
061        public boolean isWindowMode(String mode) {
062                if (mode == null) throw new IllegalArgumentException("Window mode argument cannot be null");
063                return windowMode.equals(WindowMode.valueOf(mode));
064        }
065
066        @Inject
067        private PageRenderLinkSource linkSource;
068
069        @Override
070        protected Class<?> getOauthPageClass() {
071                return TwitterOauth.class;
072        }
073
074        public String getOauthAuthorizationLink() throws TwitterException {
075                return getOauthRedirectLink(windowMode, "request_token", getReturnPageUri());
076        }
077}
