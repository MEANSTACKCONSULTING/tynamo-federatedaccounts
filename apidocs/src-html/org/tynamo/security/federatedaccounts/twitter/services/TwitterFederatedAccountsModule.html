001package org.tynamo.security.federatedaccounts.twitter.services;
002
003import org.apache.shiro.realm.AuthenticatingRealm;
004import org.apache.shiro.realm.Realm;
005import org.apache.tapestry5.ioc.Configuration;
006import org.apache.tapestry5.ioc.MappedConfiguration;
007import org.apache.tapestry5.ioc.ServiceBinder;
008import org.apache.tapestry5.ioc.annotations.Contribute;
009import org.apache.tapestry5.ioc.annotations.InjectService;
010import org.apache.tapestry5.services.LibraryMapping;
011import org.tynamo.common.ModuleProperties;
012import org.tynamo.security.federatedaccounts.FederatedAccount.FederatedAccountType;
013import org.tynamo.security.federatedaccounts.services.FederatedAccountsModule;
014import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentBlockSource;
015import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentContribution;
016import org.tynamo.security.federatedaccounts.twitter.pages.TwitterOauth;
017import org.tynamo.security.services.SecurityFilterChainFactory;
018import org.tynamo.security.services.impl.SecurityFilterChain;
019
020import twitter4j.TwitterFactory;
021
022public class TwitterFederatedAccountsModule {
023        private static final String PATH_PREFIX = "federated-twitter";
024        // this is a child module, use the same version as for the parent
025        private static String version = ModuleProperties.getVersion(FederatedAccountsModule.class);
026
027        public static void bind(ServiceBinder binder) {
028                binder.bind(AuthenticatingRealm.class, TwitterRealm.class).withId(TwitterRealm.class.getSimpleName());
029        }
030
031        public static void contributeFactoryDefaults(MappedConfiguration<String, String> configuration) {
032                configuration.add(TwitterRealm.TWITTER_PRINCIPAL, TwitterRealm.PrincipalProperty.id.name());
033                configuration.add(TwitterRealm.TWITTER_CLIENTID, "");
034                configuration.add(TwitterRealm.TWITTER_CLIENTSECRET, "");
035        }
036
037        public static void contributeComponentClassResolver(Configuration<LibraryMapping> configuration) {
038                configuration.add(new LibraryMapping(FederatedAccountsModule.PATH_PREFIX,
039                        "org.tynamo.security.federatedaccounts.twitter"));
040        }
041
042        public static void contributeClasspathAssetAliasManager(MappedConfiguration<String, String> configuration) {
043                configuration.add(PATH_PREFIX + "-" + version, "org/tynamo/security/federatedaccounts/twitter");
044        }
045
046        public static void contributeWebSecurityManager(Configuration<Realm> configuration,
047                @InjectService("TwitterRealm") AuthenticatingRealm twitterRealm) {
048                configuration.add(twitterRealm);
049        }
050
051        public static void contributeSecurityConfiguration(Configuration<SecurityFilterChain> configuration,
052                SecurityFilterChainFactory factory) {
053                configuration.add(factory
054                        .createChain("/" + FederatedAccountsModule.PATH_PREFIX + "/" + TwitterOauth.class.getSimpleName().toLowerCase())
055                        .add(factory.anon()).build());
056        }
057
058        public static TwitterFactory build() {
059                return new TwitterFactory();
060        }
061
062        @Contribute(FederatedSignInComponentBlockSource.class)
063        public static void addSignInComponentBlocks(Configuration<FederatedSignInComponentContribution> configuration) {
064                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.twitter.name(),
065                        "federated/twitterSignInComponentBlocks"));
066        }
067
068}
