001package org.tynamo.security.federatedaccounts.twitter.services;
002
003import org.apache.shiro.authc.AuthenticationException;
004import org.apache.shiro.authc.AuthenticationInfo;
005import org.apache.shiro.authc.AuthenticationToken;
006import org.apache.shiro.authc.IncorrectCredentialsException;
007import org.apache.shiro.cache.MemoryConstrainedCacheManager;
008import org.apache.shiro.realm.AuthenticatingRealm;
009import org.apache.tapestry5.ioc.annotations.Inject;
010import org.apache.tapestry5.ioc.annotations.Symbol;
011import org.slf4j.Logger;
012import org.tynamo.security.federatedaccounts.FederatedAccount;
013import org.tynamo.security.federatedaccounts.services.FederatedAccountService;
014import org.tynamo.security.federatedaccounts.twitter.TwitterAccessToken;
015
016import twitter4j.Twitter;
017import twitter4j.TwitterException;
018import twitter4j.TwitterFactory;
019import twitter4j.User;
020import twitter4j.auth.AccessToken;
021
022/**
023 * <p>
024 * A {@link org.apache.shiro.realm.Realm} that authenticates with Twitter.
025 */
026public class TwitterRealm extends AuthenticatingRealm {
027        public static final String TWITTER_CLIENTID = "twitter.clientid";
028        public static final String TWITTER_CLIENTSECRET = "twitter.clientsecret";
029        public static final String TWITTER_PRINCIPAL = "twitter.principal";
030
031        private Logger logger;
032
033        public static enum PrincipalProperty {
034                id, screenname, name
035        }
036
037        private PrincipalProperty principalProperty;
038
039        private final FederatedAccountService federatedAccountService;
040        private final TwitterFactory twitterFactory;
041        private String oauthClientId;
042        private String oauthClientSecret;
043
044        public TwitterRealm(Logger logger, FederatedAccountService federatedAccountService, TwitterFactory twitterFactory,
045                @Inject @Symbol(TwitterRealm.TWITTER_CLIENTID) String oauthClientId,
046                @Inject @Symbol(TwitterRealm.TWITTER_CLIENTSECRET) String oauthClientSecret,
047                @Inject @Symbol(TwitterRealm.TWITTER_PRINCIPAL) String principalPropertyName) {
048                super(new MemoryConstrainedCacheManager());
049                this.federatedAccountService = federatedAccountService;
050                this.twitterFactory = twitterFactory;
051                this.oauthClientId = oauthClientId;
052                this.oauthClientSecret = oauthClientSecret;
053                this.logger = logger;
054                // Let this throw IllegalArgumentException if value is not supported
055                this.principalProperty = PrincipalProperty.valueOf(principalPropertyName);
056                setName(FederatedAccount.FederatedAccountType.twitter.name());
057                setAuthenticationTokenClass(TwitterAccessToken.class);
058        }
059
060        @Override
061        protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authenticationToken)
062                throws AuthenticationException {
063                AccessToken accessToken = (AccessToken) authenticationToken.getCredentials();
064
065                Twitter twitter = twitterFactory.getInstance();
066                twitter.setOAuthConsumer(oauthClientId, oauthClientSecret);
067                twitter.setOAuthAccessToken(accessToken);
068
069                User twitterUser;
070                try {
071                        twitterUser = twitter.verifyCredentials();
072                } catch (TwitterException e) {
073                        logger.error(e.getMessage(), e);
074                        throw new IncorrectCredentialsException(
075                                "Twitter security verification failed, terminating authentication request", e);
076                }
077
078                String principalValue = null;
079                switch (principalProperty) {
080                case id:
081                        principalValue = String.valueOf(twitterUser.getId());
082                        break;
083                case screenname:
084                        principalValue = twitterUser.getScreenName();
085                        break;
086                case name:
087                        principalValue = twitterUser.getName();
088                        break;
089                }
090
091                AuthenticationInfo authenticationInfo = federatedAccountService.federate(
092                        FederatedAccount.FederatedAccountType.twitter.name(), principalValue, authenticationToken, twitterUser);
093                // returned principalcollection is immutable
094                // authenticationInfo.getPrincipals().fromRealm(FederatedAccount.Type.twitter.name()).add(authenticationToken);
095                return authenticationInfo;
096                // if (federatedAccount.isAccountLocked()) { throw new LockedAccountException("Twitter federated account ["
097                // + federatedAccount.getUsername() + "] is locked."); }
098                // if (federatedAccount.isCredentialsExpired()) {
099                // String msg = "The credentials for account [" + twitterUser.getId() + "] are expired";
100                // throw new ExpiredCredentialsException(msg);
101                // }
102                // return new SimpleAuthenticationInfo(federatedAccount.getUsername(), token.getCredentials(), getName());
103
104                // return federatedAccount;
105        }
106}
