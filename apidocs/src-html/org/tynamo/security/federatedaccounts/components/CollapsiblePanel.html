001package org.tynamo.security.federatedaccounts.components;
002
003import org.apache.tapestry5.BindingConstants;
004import org.apache.tapestry5.Block;
005import org.apache.tapestry5.annotations.Import;
006import org.apache.tapestry5.annotations.Parameter;
007import org.apache.tapestry5.annotations.Property;
008import org.apache.tapestry5.ioc.annotations.Inject;
009import org.apache.tapestry5.services.javascript.JavaScriptSupport;
010
011@Import(module = "federatedAccounts/CollapsiblePanel", stylesheet = "CollapsiblePanel.css")
012public class CollapsiblePanel {
013
014        @Inject
015        private JavaScriptSupport jss;
016
017        @Parameter(value = "defaultCollapsed", defaultPrefix = "literal")
018        private boolean collapsed;
019
020        @Parameter(defaultPrefix = "literal")
021        @Property
022        private String title;
023
024        @Parameter(defaultPrefix = BindingConstants.LITERAL)
025        @Property(write = false)
026        private Block headerContent;
027
028        @Parameter(defaultPrefix = "literal")
029        private boolean expanderDisabled;
030
031        public boolean isDefaultCollapsed() {
032                return true;
033        }
034
035        public String getState() {
036                return collapsed ? "collapsed" : "expanded";
037        }
038
039        public boolean isExpanderDisabled() {
040                return expanderDisabled;
041        }
042
043        public void setExpanderDisabled(boolean expanderDisabled) {
044                this.expanderDisabled = expanderDisabled;
045        }
046
047}
