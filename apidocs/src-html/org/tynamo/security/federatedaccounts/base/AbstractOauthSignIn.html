001package org.tynamo.security.federatedaccounts.base;
002
003import org.apache.tapestry5.BindingConstants;
004import org.apache.tapestry5.ComponentResources;
005import org.apache.tapestry5.annotations.Import;
006import org.apache.tapestry5.annotations.Parameter;
007import org.apache.tapestry5.annotations.Property;
008import org.apache.tapestry5.ioc.annotations.Inject;
009import org.apache.tapestry5.ioc.annotations.Symbol;
010import org.apache.tapestry5.services.BaseURLSource;
011import org.apache.tapestry5.services.PageRenderLinkSource;
012import org.apache.tapestry5.services.Request;
013import org.tynamo.security.federatedaccounts.FederatedAccountSymbols;
014
015@Import(library = "OauthSignIn.js")
016public abstract class AbstractOauthSignIn extends OauthComponentBase {
017        // Use ^ for current page the component is contained in, empty for configured success url
018        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "prop:defaultReturnPage")
019        String returnPageName;
020
021        public String getReturnPageName() {
022                return returnPageName;
023        }
024
025        @Inject
026        @Symbol(FederatedAccountSymbols.DEFAULT_RETURNPAGE)
027        @Property(write = false)
028        String defaultReturnPage;
029
030        @Inject
031        PageRenderLinkSource pageRenderLinkSource;
032
033        @Inject
034        ComponentResources componentResources;
035
036        @Inject
037        BaseURLSource baseURLSource;
038
039        @Inject
040        Request request;
041
042        public String getReturnPageUri() {
043                if ("^".equals(returnPageName))
044                        return pageRenderLinkSource.createPageRenderLink(componentResources.getPage().getClass()).toAbsoluteURI();
045
046                // Empty returnpageName denotes a base url. Default for default return page may be set to a non empty value.
047                // If default is empty and not locally overridden, return the baseurl
048                return "".equals(returnPageName) && "".equals(defaultReturnPage) ? baseURLSource.getBaseURL(request.isSecure())
049                        : pageRenderLinkSource.createPageRenderLink(returnPageName).toAbsoluteURI();
050
051        }
052}
