001package org.tynamo.security.federatedaccounts.facebook.components;
002
003import org.apache.tapestry5.BindingConstants;
004import org.apache.tapestry5.ComponentResources;
005import org.apache.tapestry5.annotations.Import;
006import org.apache.tapestry5.annotations.Parameter;
007import org.apache.tapestry5.annotations.Property;
008import org.apache.tapestry5.ioc.annotations.Inject;
009import org.apache.tapestry5.ioc.annotations.Symbol;
010import org.apache.tapestry5.services.AssetSource;
011import org.apache.tapestry5.services.PageRenderLinkSource;
012import org.tynamo.security.federatedaccounts.base.AbstractOauthSignIn;
013import org.tynamo.security.federatedaccounts.facebook.pages.FacebookOauth;
014import org.tynamo.security.federatedaccounts.facebook.services.FacebookRealm;
015import org.tynamo.security.federatedaccounts.util.WindowMode;
016
017@Import(stylesheet = "fb-button.css")
018public class FacebookOauthSignIn extends AbstractOauthSignIn {
019
020        @Inject
021        @Symbol(FacebookRealm.FACEBOOK_PERMISSIONS)
022        @Property
023        private String facebookPermissions;
024
025        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "800")
026        @Property
027        private Integer width;
028
029        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "400")
030        @Property
031        private Integer height;
032
033        @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "blank")
034        private WindowMode windowMode;
035
036        @Inject
037        private AssetSource assetSource;
038
039        @Inject
040        private ComponentResources componentResources;
041
042        @Inject
043        private PageRenderLinkSource pageRenderLinkSource;
044
045        public boolean isWindowMode(String mode) {
046                if (mode == null) throw new IllegalArgumentException("Window mode argument cannot be null");
047                return windowMode.equals(WindowMode.valueOf(mode));
048        }
049
050        public String getOauthAuthorizationLink() {
051                StringBuilder sb = new StringBuilder();
052                sb.append("https://graph.facebook.com/oauth/authorize?client_id=");
053                sb.append(getOauthClientId());
054                sb.append("&redirect_uri=");
055                sb.append(getOauthRedirectLink());
056                sb.append("&display=popup");
057                sb.append("&scope=");
058                sb.append(facebookPermissions);
059
060                return sb.toString();
061        }
062
063        public String getOauthRedirectLink() {
064                if ("".equals(getReturnPageName())) return getOauthRedirectLink(windowMode);
065                if ("^".equals(getReturnPageName()))
066                        return getOauthRedirectLink(windowMode,
067                                pageRenderLinkSource.createPageRenderLink(componentResources.getPage().getClass()).toAbsoluteURI());
068                return getOauthRedirectLink(windowMode, pageRenderLinkSource.createPageRenderLink(getReturnPageName())
069                        .toAbsoluteURI());
070        }
071
072        @Override
073        protected Class<?> getOauthPageClass() {
074                return FacebookOauth.class;
075        }
076
077
078}
