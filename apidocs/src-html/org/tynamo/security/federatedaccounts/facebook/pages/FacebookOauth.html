001package org.tynamo.security.federatedaccounts.facebook.pages;
002
003import java.net.MalformedURLException;
004import java.net.URI;
005import java.util.ArrayList;
006import java.util.List;
007
008import org.apache.http.HttpEntity;
009import org.apache.http.HttpResponse;
010import org.apache.http.HttpStatus;
011import org.apache.http.NameValuePair;
012import org.apache.http.client.HttpClient;
013import org.apache.http.client.methods.HttpGet;
014import org.apache.http.client.utils.URIUtils;
015import org.apache.http.client.utils.URLEncodedUtils;
016import org.apache.http.impl.client.DefaultHttpClient;
017import org.apache.http.message.BasicNameValuePair;
018import org.apache.http.util.EntityUtils;
019import org.apache.shiro.SecurityUtils;
020import org.apache.shiro.authc.AuthenticationException;
021import org.apache.tapestry5.EventContext;
022import org.apache.tapestry5.alerts.AlertManager;
023import org.apache.tapestry5.ioc.annotations.Inject;
024import org.apache.tapestry5.services.PageRenderLinkSource;
025import org.apache.tapestry5.services.Request;
026import org.slf4j.Logger;
027import org.tynamo.security.federatedaccounts.base.AbstractOauthPage;
028import org.tynamo.security.federatedaccounts.facebook.FacebookAccessToken;
029
030public class FacebookOauth extends AbstractOauthPage {
031        @Inject
032        private Logger logger;
033
034        @Inject
035        private Request request;
036
037        @Inject
038        private AlertManager alertManager;
039
040        @Inject
041        private PageRenderLinkSource linkSource;
042
043        protected Object onOauthActivate(EventContext eventContext) throws MalformedURLException {
044                String code = request.getParameter("code");
045
046                if (eventContext.getCount() > 1) setReturnUri(eventContext.get(String.class, 1));
047
048                if (code == null) {
049                        alertManager.error("No Oauth authentication code provided");
050                        return null;
051                }
052
053                List<NameValuePair> qparams = new ArrayList<NameValuePair>();
054                qparams.add(new BasicNameValuePair("client_id", getOauthClientId()));
055                qparams.add(new BasicNameValuePair("redirect_uri", getOauthRedirectLink(getWindowMode(), getReturnUri())));
056                qparams.add(new BasicNameValuePair("client_secret", getOauthClientSecret()));
057                qparams.add(new BasicNameValuePair("code", code));
058                HttpGet get = null;
059                String accessToken = "";
060                long expires = 0L;
061                try {
062                        URI uri = URIUtils.createURI("https", "graph.facebook.com", -1, "/oauth/access_token",
063                                URLEncodedUtils.format(qparams, "UTF-8"), null);
064                        get = new HttpGet(uri);
065
066                        // HttpGet get = new HttpGet(accessTokenUri);
067                        // NameValuePair[] queryString = new NameValuePair[4];
068                        // queryString[0] = new NameValuePair("client_id", clientId);
069                        // queryString[1] = new NameValuePair("redirect_uri", getOauthRedirectLink());
070                        // queryString[2] = new NameValuePair("client_secret", clientSecret);
071                        // queryString[3] = new NameValuePair("code", code);
072                        // get.setQueryString(queryString);
073                        // HttpClient httpClient = new DefaultHttpClient();
074                        HttpClient httpClient = new DefaultHttpClient();
075
076                        HttpResponse response = httpClient.execute(get);
077                        int status = response.getStatusLine().getStatusCode();
078                        if (HttpStatus.SC_OK != status) {
079                                logger.error("Facebook access_token request returned status code " + status);
080                                alertManager.error("Facebook access_token request failed with status code: " + status);
081                                return null;
082                        }
083                        HttpEntity entity = response.getEntity();
084                        if (entity != null) {
085                                long len = entity.getContentLength();
086                                if (len != -1 && len < 2048) accessToken = EntityUtils.toString(entity);
087                        }
088                } catch (Exception e) {
089                        logger.error("Facebook access_token request failed because of:", e);
090                        alertManager.error("Facebook access_token request failed with message: " + e.getMessage());
091                        return null;
092                } finally {
093                        if (get != null) get.abort();
094                }
095
096                try {
097                        if (!accessToken.startsWith("access_token")) throw new IllegalArgumentException();
098                        // access_token is of form:
099                        // access_token=119507274749030|2.1AptZFp9__qW3k2PuG4bVA__.3600.1274914800-539598633|9aTyryhVl8vnn3ulLy2w6Txo92E.&expires=4059
100                        accessToken = accessToken.substring(accessToken.indexOf("=") + 1);
101                        expires = Long.valueOf(accessToken.substring(accessToken.lastIndexOf("=") + 1));
102                        accessToken = accessToken.substring(0, accessToken.indexOf("&expires"));
103                } catch (Exception e) {
104                        logger.error("access_token wasn't of right format");
105                        alertManager.error("Facebook access_token wasn't of right format");
106                        return null;
107                }
108
109                FacebookAccessToken fbAccessToken = new FacebookAccessToken(accessToken, expires);
110                fbAccessToken.setRememberMe(isRememberMe());
111                try {
112                        SecurityUtils.getSubject().login(fbAccessToken);
113                        alertManager.success("User successfully authenticated");
114                        setOauthAuthenticated(true);
115                } catch (AuthenticationException e) {
116                        logger.error("Using access token " + accessToken + "\nCould not sign in a Facebook federated user because of: ",
117                                e);
118                        // FIXME Deal with other account exception types like expired and
119                        // locked
120                        alertManager.error("A Facebook federated user cannot be signed in, report this to support.\n "
121                                + e.getMessage());
122                        return null;
123                }
124                return null;
125        }
126}
