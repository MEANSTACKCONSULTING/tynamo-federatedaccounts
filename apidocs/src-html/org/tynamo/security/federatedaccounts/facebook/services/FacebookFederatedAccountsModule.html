001package org.tynamo.security.federatedaccounts.facebook.services;
002
003import org.apache.shiro.realm.AuthenticatingRealm;
004import org.apache.shiro.realm.Realm;
005import org.apache.tapestry5.ioc.Configuration;
006import org.apache.tapestry5.ioc.MappedConfiguration;
007import org.apache.tapestry5.ioc.ServiceBinder;
008import org.apache.tapestry5.ioc.annotations.Contribute;
009import org.apache.tapestry5.ioc.annotations.InjectService;
010import org.apache.tapestry5.services.LibraryMapping;
011import org.tynamo.common.ModuleProperties;
012import org.tynamo.security.federatedaccounts.FederatedAccount.FederatedAccountType;
013import org.tynamo.security.federatedaccounts.facebook.pages.FacebookOauth;
014import org.tynamo.security.federatedaccounts.services.FederatedAccountsModule;
015import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentBlockSource;
016import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentContribution;
017import org.tynamo.security.services.SecurityFilterChainFactory;
018import org.tynamo.security.services.impl.SecurityFilterChain;
019
020public class FacebookFederatedAccountsModule {
021        private static final String PATH_PREFIX = "federated-facebook";
022        private static String version = ModuleProperties.getVersion(FacebookFederatedAccountsModule.class);
023
024        public static void bind(ServiceBinder binder) {
025                binder.bind(AuthenticatingRealm.class, FacebookRealm.class).withId(FacebookRealm.class.getSimpleName());
026        }
027
028        public static void contributeFactoryDefaults(MappedConfiguration<String, String> configuration) {
029                configuration.add(FacebookRealm.FACEBOOK_PRINCIPAL, FacebookRealm.PrincipalProperty.id.name());
030                configuration.add(FacebookRealm.FACEBOOK_PERMISSIONS, "");
031                configuration.add(FacebookRealm.FACEBOOK_CLIENTID, "");
032                configuration.add(FacebookRealm.FACEBOOK_CLIENTSECRET, "");
033        }
034
035        public static void contributeComponentClassResolver(Configuration<LibraryMapping> configuration) {
036                configuration.add(new LibraryMapping(FederatedAccountsModule.PATH_PREFIX,
037                        "org.tynamo.security.federatedaccounts.facebook"));
038        }
039
040//      public static void contributeClasspathAssetAliasManager(MappedConfiguration<String, String> configuration) {
041//              configuration.add(PATH_PREFIX + "-" + version, "org/tynamo/security/federatedaccounts/facebook");
042//      }
043
044        public static void contributeWebSecurityManager(Configuration<Realm> configuration,
045                        @InjectService("FacebookRealm") AuthenticatingRealm facebookRealm) {
046                configuration.add(facebookRealm);
047        }
048
049        public static void contributeSecurityConfiguration(Configuration<SecurityFilterChain> configuration,
050                        SecurityFilterChainFactory factory) {
051                configuration.add(factory
052                        .createChain("/" + FederatedAccountsModule.PATH_PREFIX + "/"
053                                + FacebookOauth.class.getSimpleName().toLowerCase()).add(factory.anon()).build());
054        }
055
056        @Contribute(FederatedSignInComponentBlockSource.class)
057        public static void addSignInComponentBlocks(Configuration<FederatedSignInComponentContribution> configuration) {
058                configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.facebook.name(),
059                        "federated/facebookSignInComponentBlocks"));
060        }
061
062}
