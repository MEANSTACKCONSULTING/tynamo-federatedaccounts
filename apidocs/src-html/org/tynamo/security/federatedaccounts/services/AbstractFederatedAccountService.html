001package org.tynamo.security.federatedaccounts.services;
002
003import java.util.Map;
004
005import org.apache.shiro.authc.AuthenticationException;
006import org.apache.shiro.authc.AuthenticationInfo;
007import org.apache.shiro.authc.AuthenticationToken;
008import org.apache.shiro.authc.ExpiredCredentialsException;
009import org.apache.shiro.authc.LockedAccountException;
010import org.apache.shiro.authc.SimpleAuthenticationInfo;
011import org.apache.shiro.subject.SimplePrincipalCollection;
012import org.apache.tapestry5.ioc.annotations.Symbol;
013import org.slf4j.Logger;
014import org.tynamo.security.federatedaccounts.FederatedAccount;
015import org.tynamo.security.federatedaccounts.FederatedAccountSymbols;
016
017public abstract class AbstractFederatedAccountService implements FederatedAccountService {
018
019        protected final Map<String, Object> entityTypesByRealm;
020        protected final Class<?> singleEntityType;
021        protected final Logger logger;
022        protected final String localAccountRealmName;
023
024        public AbstractFederatedAccountService(Logger logger,
025                @Symbol(FederatedAccountSymbols.LOCALACCOUNT_REALMNAME) String localAccountRealmName,
026                Map<String, Object> entityTypesByRealm) {
027                this.logger = logger;
028                this.localAccountRealmName = localAccountRealmName;
029                this.entityTypesByRealm = entityTypesByRealm;
030                singleEntityType = entityTypesByRealm.containsKey("*") ? (Class<?>) entityTypesByRealm.get("*") : null;
031        }
032
033        protected final Map<String, Object> getEntityTypesByRealm() {
034                return entityTypesByRealm;
035        }
036
037        protected FederatedAccount createLocalAccount(Class<?> entityType, String realmName, Object remotePrincipal,
038                Object remoteAccount) {
039                // You could also merge the account if there's already an existing user with the same email/username as
040                // in the remoteAccount
041                FederatedAccount localAccount = null;
042                try {
043                        localAccount = (FederatedAccount) entityType.newInstance();
044                        // InstantiationException, IllegalAccessException
045                } catch (Exception e) {
046                        throw new AuthenticationException("Entity of type " + localAccount.getClass().getSimpleName()
047                                + " is not an instance of " + FederatedAccount.class.getSimpleName(), e);
048                }
049                localAccount.federate(realmName, remotePrincipal, remoteAccount);
050                // new account, always save
051                saveAccount(localAccount);
052                return localAccount;
053        }
054
055        protected abstract void saveAccount(FederatedAccount account);
056
057        protected boolean updateLocalAccount(FederatedAccount localUser, String realmName, Object remotePrincipal,
058                Object remoteAccount) {
059                boolean modified = localUser.federate(realmName, null, remoteAccount);
060                if (modified) updateAccount(localUser);
061                return modified;
062        }
063
064        protected abstract void updateAccount(FederatedAccount account);
065
066        protected abstract FederatedAccount findLocalAccount(Class<?> entityType, String realmName, Object remotePrincipal,
067                Object remoteAccount);
068
069        public AuthenticationInfo federate(String realmName, Object remotePrincipal, AuthenticationToken authenticationToken,
070                Object remoteAccount) {
071                Class<?> entityType = singleEntityType == null ? (Class<?>) entityTypesByRealm.get(realmName) : singleEntityType;
072                if (entityType == null) return null;
073
074                FederatedAccount localAccount = findLocalAccount(entityType, realmName, remotePrincipal, remoteAccount);
075                if (localAccount == null) localAccount = createLocalAccount(entityType, realmName, remotePrincipal, remoteAccount);
076                else updateLocalAccount(localAccount, realmName, null, remoteAccount);
077
078                if (localAccount.isAccountLocked()) { throw new LockedAccountException("Federated account [" + remotePrincipal
079                        + "] is locked."); }
080                if (localAccount.isCredentialsExpired()) {
081                        String msg = "The credentials for federated account [" + remotePrincipal + "] are expired";
082                        throw new ExpiredCredentialsException(msg);
083                }
084                SimplePrincipalCollection principalCollection;
085                if (localAccountRealmName.isEmpty()) principalCollection = new SimplePrincipalCollection(remotePrincipal, realmName);
086                else {
087                        principalCollection = new SimplePrincipalCollection(localAccount.getLocalAccountPrimaryPrincipal(),
088                                localAccountRealmName);
089                        principalCollection.add(remotePrincipal, realmName);
090                }
091                principalCollection.add(authenticationToken, realmName);
092                return new SimpleAuthenticationInfo(principalCollection, authenticationToken.getCredentials());
093
094                // SimpleAuthenticationInfo info = new SimpleAuthenticationInfo(remotePrincipal,
095                // authenticationToken.getCredentials(), realmName);
096        }
097}
