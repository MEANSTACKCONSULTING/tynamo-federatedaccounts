001package org.tynamo.security.federatedaccounts.services;
002
003import org.apache.tapestry5.ioc.internal.util.InternalUtils;
004import org.apache.tapestry5.services.BeanBlockSource;
005
006/**
007 * A contribution to the {@link BeanBlockSource} service, defining a page name and block id (within the page) that can edit or display a
008 * particular type of property.
009 */
010public class FederatedSignInComponentContribution {
011        private final String key;
012
013        private final String pageName;
014
015        public FederatedSignInComponentContribution(String key, String pageName) {
016                assert InternalUtils.isNonBlank(key);
017                assert InternalUtils.isNonBlank(pageName);
018                this.key = key;
019                this.pageName = pageName;
020        }
021
022        /**
023         * The type of data for which the indicated block will provide an editor or displayer for.
024         */
025        public final String getKey() {
026                return key;
027        }
028
029        /**
030         * The logical name of the page containing the block.
031         */
032        public final String getPageName() {
033                return pageName;
034        }
035
036}
