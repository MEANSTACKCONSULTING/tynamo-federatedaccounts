001package org.tynamo.security.federatedaccounts.services;
002
003import org.apache.tapestry5.ioc.Configuration;
004import org.apache.tapestry5.ioc.MappedConfiguration;
005import org.apache.tapestry5.ioc.ServiceBinder;
006import org.apache.tapestry5.ioc.services.Coercion;
007import org.apache.tapestry5.ioc.services.CoercionTuple;
008import org.apache.tapestry5.services.LibraryMapping;
009import org.tynamo.common.ModuleProperties;
010import org.tynamo.security.federatedaccounts.FederatedAccountSymbols;
011import org.tynamo.security.federatedaccounts.util.WindowMode;
012
013public class FederatedAccountsModule {
014        public static final String PATH_PREFIX = "federated";
015        private static String version = ModuleProperties.getVersion(FederatedAccountsModule.class);
016
017        public static void bind(ServiceBinder binder) {
018                binder.bind(FederatedSignInComponentBlockSource.class, FederatedSignInComponentBlockSourceImpl.class);
019                binder.bind(FederatedSignInOptions.class, FederatedSignInOptionsImpl.class);
020        }
021
022        public static void contributeFactoryDefaults(MappedConfiguration<String, String> configuration) {
023                configuration.add(FederatedAccountSymbols.COMMITAFTER_OAUTH, "true");
024                configuration.add(FederatedAccountSymbols.HTTPCLIENT_ON_GAE, "false");
025                configuration.add(FederatedAccountSymbols.DEFAULT_RETURNPAGE, "");
026                configuration.add(FederatedAccountSymbols.DEFAULT_REMEMBERME, "true");
027                configuration.add(FederatedAccountSymbols.LOCALACCOUNT_REALMNAME, "");
028        }
029
030        public static void contributeComponentClassResolver(Configuration<LibraryMapping> configuration) {
031                configuration.add(new LibraryMapping(PATH_PREFIX, "org.tynamo.security.federatedaccounts"));
032        }
033
034//      public static void contributeClasspathAssetAliasManager(MappedConfiguration<String, String> configuration) {
035//              configuration.add(PATH_PREFIX + "-" + version, "org/tynamo/security/federatedaccounts");
036//      }
037
038        public void contributeTypeCoercer(Configuration<CoercionTuple<String, WindowMode>> configuration) {
039                configuration.add(new CoercionTuple<String, WindowMode>(String.class, WindowMode.class,
040                        new Coercion<String, WindowMode>() {
041                                public WindowMode coerce(String input) {
042                                        return WindowMode.valueOf(input);
043                                }
044                        }));
045        }
046
047}
