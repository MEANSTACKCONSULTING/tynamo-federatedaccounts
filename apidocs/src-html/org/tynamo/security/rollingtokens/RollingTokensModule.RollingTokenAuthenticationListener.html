091        public static class RollingTokenAuthenticationListener implements CommittingAuthenticationListener {
092                private static Random random = new Random((new Date()).getTime());
093
094                // TODO we should probably allow supplying this as a symbol
095                // In seconds, default is a month
096                private int maxAge = 30 * 24 * 3600;
097
098                private final HttpServletRequest request;
099                private final HttpServletResponse response;
100                private final Cookies cookies;
101                private final EntityManager entityManager;
102                private String realmName;
103                private Class principalType;
104
105                public RollingTokenAuthenticationListener(EntityManager entityManager, Cookies cookies, HttpServletRequest request,
106                        HttpServletResponse response, @Inject @Symbol(RollingTokenSymbols.CONFIGURED_REALM) String realmName,
107                        @Inject @Symbol(RollingTokenSymbols.CONFIGURED_PRINCIPALTYPE) String principalType) throws ClassNotFoundException {
108                        this.entityManager = entityManager;
109                        this.cookies = cookies;
110                        this.request = request;
111                        this.response = response;
112                        this.realmName = realmName;
113                        this.principalType = principalType.isEmpty() ? null : Class.forName(principalType);
114                }
115
116                @Override
117                public void onLogout(PrincipalCollection principals) {
118                        eraseAllExpiringRollingTokens(principals.byType(RollingToken.class));
119                        cookies.removeCookieValue(RollingToken.TOKEN_NAME);
120                }
121
122                public void onFailure(AuthenticationToken token, AuthenticationException ae) {
123                        if (token instanceof RollingToken) {
124                                if (token.getPrincipal() == null || token.getCredentials() == null) return;
125                                // The token might still exist after it's expired, so clean up
126                                // ExpiringRollingToken expiringToken = (ExpiringRollingToken) session.createCriteria(ExpiringRollingToken.class)
127                                // .add(Restrictions.eq("name", token.getPrincipal().toString())).add(Restrictions.eq("value",
128                                // token.getCredentials().toString()))
129                                // .uniqueResult();
130                                // if (expiringToken != null) session.delete(expiringToken);
131                                Query query = entityManager.createQuery("delete from " + ExpiringRollingToken.class.getSimpleName()
132                                        + " e where e.name = ?1 and e.value = ?2");
133                                query.setParameter(1, token.getPrincipal().toString());
134                                query.setParameter(2, token.getCredentials().toString());
135                                query.executeUpdate();
136
137                                // If this is rememberme authentication, security is run before request and cookie shadows are established
138                                // so use native servlet request here
139                                // cookies.removeCookieValue(UsernameRollingToken.TOKEN_NAME);
140                                Cookie cookie = new Cookie(RollingToken.TOKEN_NAME, "xxx");
141                                cookie.setMaxAge(0);
142                                response.addCookie(cookie);
143                        }
144                }
145
146                @Override
147                public void onSuccess(AuthenticationToken token, AuthenticationInfo info) {
148                        // Note that the same token is actually rolling - it creates a new token and "expires" the old one
149                        // by replacing it. This *may* cause problems and even an infinite loop if you are not careful -
150                        // with a single refresh, the old token is sent several times with each request
151                        if (token instanceof RollingToken) return;
152                        if (!(token instanceof RememberMeAuthenticationToken)) return;
153                        RememberMeAuthenticationToken rememberMeToken = (RememberMeAuthenticationToken) token;
154                        if (rememberMeToken.isRememberMe()) {
155                                String tokenValue = (new BigInteger(128, random)).toString();
156                                Date expirationDate = new Date((new Date()).getTime() + maxAge * 1000L);
157
158                                CriteriaBuilder builder = entityManager.getCriteriaBuilder();
159                                CriteriaQuery<ExpiringRollingToken> criteriaQuery = builder.createQuery(ExpiringRollingToken.class);
160                                Root<ExpiringRollingToken> from = criteriaQuery.from(ExpiringRollingToken.class);
161                                Object principal = RollingToken.getConfiguredPrincipal(realmName, principalType, info.getPrincipals());
162                                Predicate predicate = builder.and(builder.equal(from.get("name"), principal.toString()),
163                                        builder.equal(from.get("hostAddress"), request.getRemoteAddr()));
164                                ExpiringRollingToken expiringRollingToken = null;
165                                List<ExpiringRollingToken> results = entityManager.createQuery(criteriaQuery.where(predicate)).getResultList();
166
167                                if (results.size() <= 0) expiringRollingToken = new ExpiringRollingToken(principal.toString(), tokenValue,
168                                        request.getRemoteAddr(), expirationDate);
169                                else {
170                                        expiringRollingToken = results.remove(0);
171                                        // there shouldn't be any remaining, but remove the rest in case there are
172                                        for (ExpiringRollingToken obsoleteToken : results)
173                                                entityManager.remove(obsoleteToken);
174                                        expiringRollingToken.setExpiresAfter(expirationDate);
175                                        expiringRollingToken.setValue(tokenValue);
176                                }
177                                entityManager.persist(expiringRollingToken);
178                                cookies.writeCookieValue(RollingToken.TOKEN_NAME, tokenValue);
179
180                        }
181                }

183                // TODO Should the argument be Collection<Object> principals instead?
184                protected void eraseAllExpiringRollingTokens(Collection<RollingToken> rollingTokens) {
185                        // List<ExpiringRollingToken> rollingTokens = session.createCriteria(ExpiringRollingToken.class)
186                        // .add(Restrictions.eq("name", principal.toString())).list();
187                        // for (ExpiringRollingToken token : rollingTokens)
188                        // session.delete(token);
189                        Query query = entityManager.createQuery("delete from " + ExpiringRollingToken.class.getSimpleName()
190                                + " where name = ?1");
191                        for (RollingToken token : rollingTokens) {
192                                query.setParameter(1, token.getPrincipal().toString());
193                                query.executeUpdate();
194                        }
195                }
