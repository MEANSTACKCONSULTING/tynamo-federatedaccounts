001package org.tynamo.security.rollingtokens.services;
002
003import java.util.Collection;
004
005import org.apache.shiro.authc.AuthenticationToken;
006import org.apache.shiro.authc.HostAuthenticationToken;
007import org.apache.shiro.authc.RememberMeAuthenticationToken;
008import org.apache.shiro.subject.PrincipalCollection;
009
010public class RollingToken implements AuthenticationToken, HostAuthenticationToken, RememberMeAuthenticationToken {
011        public static final String TOKEN_NAME = "rollingToken";
012
013        private Object principal;
014        private String rollingToken;
015        private String hostAddress;
016
017        public RollingToken(Object principal, String rollingToken, String remoteAddress) {
018                this.principal = principal;
019                this.rollingToken = rollingToken;
020                hostAddress = remoteAddress;
021        }
022
023        /**
024         * 
025         */
026        private static final long serialVersionUID = -3054718345404435729L;
027
028        public Object getCredentials() {
029                return rollingToken;
030        }
031
032        public Object getPrincipal() {
033                return principal;
034        }
035
036        public boolean isRememberMe() {
037                // Always return true, otherwise rememberMe token is erased when this authentication is used
038                return true;
039        }
040
041        public String getHost() {
042                return hostAddress;
043        }
044
045        // TODO the implementation is almost the same as in tapestry-security-jpa's SecureEntityManager.getConfiguredPrincipal
046        // is there a possibility to refactor this to some common class?
047        public static Object getConfiguredPrincipal(String realmName, Class<?> principalType,
048                PrincipalCollection allPrincipals) {
049                if (!realmName.isEmpty()) {
050                        Collection<?> principals = allPrincipals.fromRealm(realmName);
051                        if (principalType == null) principals.iterator().next();
052                        else {
053                                for (Object availablePrincipal : principals)
054                                        if (availablePrincipal.getClass().isAssignableFrom(principalType)) { return availablePrincipal; }
055                        }
056                }
057                if (principalType != null) {
058                        Object principal = allPrincipals.oneByType(principalType);
059                        if (principal == null)
060                                throw new NullPointerException("Subject is required to have a configured principal of type '" + principalType);
061                        return principal;
062                }
063                return allPrincipals.getPrimaryPrincipal();
064        }
065
066}
