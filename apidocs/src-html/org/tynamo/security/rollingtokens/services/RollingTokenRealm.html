001package org.tynamo.security.rollingtokens.services;
002
003import java.util.Date;
004import java.util.List;
005
006import javax.persistence.EntityManager;
007import javax.persistence.criteria.CriteriaBuilder;
008import javax.persistence.criteria.CriteriaQuery;
009import javax.persistence.criteria.Predicate;
010import javax.persistence.criteria.Root;
011
012import org.apache.shiro.authc.AuthenticationException;
013import org.apache.shiro.authc.AuthenticationInfo;
014import org.apache.shiro.authc.AuthenticationToken;
015import org.apache.shiro.realm.AuthenticatingRealm;
016import org.slf4j.Logger;
017import org.tynamo.security.federatedaccounts.services.FederatedAccountService;
018import org.tynamo.security.rollingtokens.entities.ExpiringRollingToken;
019
020// instead of extending, this should be a delegating realm, requiring the primary realm so it can query
021// the realm name and use its authorization if it so wishes
022public class RollingTokenRealm extends AuthenticatingRealm {
023
024        public static final String NAME = "rollingtokens";
025        private final EntityManager entityManager;
026        private final Logger logger;
027        private FederatedAccountService federatedAccountService;
028
029        public RollingTokenRealm(Logger logger, EntityManager entityManager, FederatedAccountService federatedAccountService) {
030                this.logger = logger;
031                this.entityManager = entityManager;
032                this.federatedAccountService = federatedAccountService;
033                setName(NAME);
034                this.setAuthenticationTokenClass(RollingToken.class);
035        }
036
037        @Override
038        protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
039                // FIXME this probably shouldn't silently fail - it seems to invalidate even the existing subject
040                RollingToken upToken = (RollingToken) token;
041
042                Object principal = token.getPrincipal();
043
044                // Null username is invalid
045                if (principal == null) {
046                        logger.warn("Null usernames are not allowed by this realm.");
047                        return null;
048                }
049
050                String hostAddress = upToken.getHost();
051                CriteriaBuilder builder = entityManager.getCriteriaBuilder();
052                CriteriaQuery<ExpiringRollingToken> criteriaQuery = builder.createQuery(ExpiringRollingToken.class);
053                Root<ExpiringRollingToken> from = criteriaQuery.from(ExpiringRollingToken.class);
054
055                Predicate predicate = builder.and(builder.equal(from.get("name"), principal.toString()),
056                        builder.greaterThan(from.<Date> get("expiresAfter"), new Date()),
057                        builder.equal(from.get("hostAddress"), hostAddress));
058
059                List<ExpiringRollingToken> rollingTokens = entityManager.createQuery(criteriaQuery.where(predicate))
060                        .getResultList();
061
062                if (rollingTokens.size() <= 0) return null;
063
064                // FIXME might need to pass a list of rolling tokens
065                return federatedAccountService.federate(getName(), principal, token, null);
066        }
067
068}
