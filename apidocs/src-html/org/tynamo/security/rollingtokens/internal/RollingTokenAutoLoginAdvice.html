001package org.tynamo.security.rollingtokens.internal;
002
003import javax.servlet.http.Cookie;
004import javax.servlet.http.HttpServletRequest;
005
006import org.apache.shiro.authc.AuthenticationException;
007import org.apache.shiro.subject.PrincipalCollection;
008import org.apache.shiro.subject.Subject;
009import org.apache.tapestry5.plastic.MethodAdvice;
010import org.apache.tapestry5.plastic.MethodInvocation;
011import org.slf4j.Logger;
012import org.tynamo.security.rollingtokens.services.RollingToken;
013
014public class RollingTokenAutoLoginAdvice implements MethodAdvice {
015        private final Logger logger;
016        private final HttpServletRequest request;
017        private String realmName;
018        private Class<?> principalType;
019
020        public RollingTokenAutoLoginAdvice(Logger logger, HttpServletRequest request, String realmName, Class<?> principalType)
021                throws ClassNotFoundException {
022                this.logger = logger;
023                this.request = request;
024                this.realmName = realmName;
025                this.principalType = principalType;
026        }
027
028        @Override
029        public void advise(MethodInvocation invocation) {
030                invocation.proceed();
031                // SubjectContext context = (SubjectContext) invocation.getParameter(0);
032                // if (!(context instanceof WebSubjectContext)) { return; }
033                // WebSubjectContext wsc = (WebSubjectContext) context;
034                // PrincipalCollection principals = wsc.resolvePrincipals();
035                // boolean authenticated = wsc.resolveAuthenticated();
036                Subject subject = (Subject) invocation.getReturnValue();
037                PrincipalCollection principals = subject.getPrincipals();
038                if (principals == null || subject.isAuthenticated()) return;
039
040                String rollingTokenValue = null;
041                try {
042                        // one issue is that underlying request shadow may at times be null
043                        // A bigger problem is that when security is decorating the request, it needs a subject
044                        // but we are only building it here. So we can't use Request shadow
045                        Cookie[] cookies = request.getCookies();
046                        // System.out.println("request " + request + " " + request.getCookies());
047                        for (Cookie cookie : cookies)
048                                if (RollingToken.TOKEN_NAME.equals(cookie.getName())) {
049                                        rollingTokenValue = cookie.getValue();
050                                        break;
051                                }
052                } catch (RuntimeException e) {
053                        // ignore
054                }
055                if (rollingTokenValue == null) return;
056
057                Object principal = RollingToken.getConfiguredPrincipal(realmName, principalType, principals);
058
059                RollingToken token = new RollingToken(principal, rollingTokenValue, request.getRemoteAddr());
060                try {
061                        subject.login(token);
062                        logger.info("Rolling token authentication using token " + rollingTokenValue
063                                + " succeeded for user identified by " + principal);
064                } catch (AuthenticationException e) {
065                        logger.info("Rolling token authentication using token " + rollingTokenValue + " failed for user identified by "
066                                + principal);
067                }
068
069        }
070
071}
