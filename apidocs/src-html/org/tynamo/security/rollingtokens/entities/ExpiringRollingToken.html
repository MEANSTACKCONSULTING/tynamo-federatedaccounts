001package org.tynamo.security.rollingtokens.entities;
002
003import java.util.Date;
004
005import javax.persistence.Entity;
006import javax.persistence.GeneratedValue;
007import javax.persistence.GenerationType;
008import javax.persistence.Id;
009import javax.persistence.Temporal;
010import javax.persistence.TemporalType;
011
012@Entity
013// FIXME Look into combining both ExpiringRollingToken and UsernameRollingToken into one
014public class ExpiringRollingToken {
015        private Long id;
016        private String name;
017        private String value;
018        private Date expiresAfter = new Date();
019        private String hostAddress;
020
021        public ExpiringRollingToken() {
022        }
023
024        public ExpiringRollingToken(String name, String value, String hostAddress, Date expiresAfter) {
025                this();
026                this.name = name;
027                this.value = value;
028                this.hostAddress = hostAddress;
029                setExpiresAfter(expiresAfter);
030        }
031
032        @Id
033        @GeneratedValue(strategy = GenerationType.AUTO)
034        public Long getId() {
035                return id;
036        }
037
038        public void setId(Long id) {
039                this.id = id;
040        }
041
042        public String getValue() {
043                return value;
044        }
045
046        public void setValue(String token) {
047                this.value = token;
048        }
049
050        public String getName() {
051                return name;
052        }
053
054        public void setName(String name) {
055                this.name = name;
056        }
057
058        @Temporal(TemporalType.TIMESTAMP)
059        public Date getExpiresAfter() {
060                return expiresAfter;
061        }
062
063        public void setExpiresAfter(Date expiresAfter) {
064                if (expiresAfter != null) this.expiresAfter = expiresAfter;
065        }
066
067        public void setHostAddress(String hostAddress) {
068                this.hostAddress = hostAddress;
069        }
070
071        public String getHostAddress() {
072                return hostAddress;
073        }
074}
