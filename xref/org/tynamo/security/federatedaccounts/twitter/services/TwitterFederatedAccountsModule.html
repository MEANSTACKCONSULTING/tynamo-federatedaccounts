1   package org.tynamo.security.federatedaccounts.twitter.services;
2   
3   import org.apache.shiro.realm.AuthenticatingRealm;
4   import org.apache.shiro.realm.Realm;
5   import org.apache.tapestry5.ioc.Configuration;
6   import org.apache.tapestry5.ioc.MappedConfiguration;
7   import org.apache.tapestry5.ioc.ServiceBinder;
8   import org.apache.tapestry5.ioc.annotations.Contribute;
9   import org.apache.tapestry5.ioc.annotations.InjectService;
10  import org.apache.tapestry5.services.LibraryMapping;
11  import org.tynamo.common.ModuleProperties;
12  import org.tynamo.security.federatedaccounts.FederatedAccount.FederatedAccountType;
13  import org.tynamo.security.federatedaccounts.services.FederatedAccountsModule;
14  import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentBlockSource;
15  import org.tynamo.security.federatedaccounts.services.FederatedSignInComponentContribution;
16  import org.tynamo.security.federatedaccounts.twitter.pages.TwitterOauth;
17  import org.tynamo.security.services.SecurityFilterChainFactory;
18  import org.tynamo.security.services.impl.SecurityFilterChain;
19  
20  import twitter4j.TwitterFactory;
21  
22  public class TwitterFederatedAccountsModule {
23  	private static final String PATH_PREFIX = "federated-twitter";
24  	// this is a child module, use the same version as for the parent
25  	private static String version = ModuleProperties.getVersion(FederatedAccountsModule.class);
26  
27  	public static void bind(ServiceBinder binder) {
28  		binder.bind(AuthenticatingRealm.class, TwitterRealm.class).withId(TwitterRealm.class.getSimpleName());
29  	}
30  
31  	public static void contributeFactoryDefaults(MappedConfiguration<String, String> configuration) {
32  		configuration.add(TwitterRealm.TWITTER_PRINCIPAL, TwitterRealm.PrincipalProperty.id.name());
33  		configuration.add(TwitterRealm.TWITTER_CLIENTID, "");
34  		configuration.add(TwitterRealm.TWITTER_CLIENTSECRET, "");
35  	}
36  
37  	public static void contributeComponentClassResolver(Configuration<LibraryMapping> configuration) {
38  		configuration.add(new LibraryMapping(FederatedAccountsModule.PATH_PREFIX,
39  			"org.tynamo.security.federatedaccounts.twitter"));
40  	}
41  
42  	public static void contributeClasspathAssetAliasManager(MappedConfiguration<String, String> configuration) {
43  		configuration.add(PATH_PREFIX + "-" + version, "org/tynamo/security/federatedaccounts/twitter");
44  	}
45  
46  	public static void contributeWebSecurityManager(Configuration<Realm> configuration,
47  		@InjectService("TwitterRealm") AuthenticatingRealm twitterRealm) {
48  		configuration.add(twitterRealm);
49  	}
50  
51  	public static void contributeSecurityConfiguration(Configuration<SecurityFilterChain> configuration,
52  		SecurityFilterChainFactory factory) {
53  		configuration.add(factory
54  			.createChain("/" + FederatedAccountsModule.PATH_PREFIX + "/" + TwitterOauth.class.getSimpleName().toLowerCase())
55  			.add(factory.anon()).build());
56  	}
57  
58  	public static TwitterFactory build() {
59  		return new TwitterFactory();
60  	}
61  
62  	@Contribute(FederatedSignInComponentBlockSource.class)
63  	public static void addSignInComponentBlocks(Configuration<FederatedSignInComponentContribution> configuration) {
64  		configuration.add(new FederatedSignInComponentContribution(FederatedAccountType.twitter.name(),
65  			"federated/twitterSignInComponentBlocks"));
66  	}
67  
68  }
