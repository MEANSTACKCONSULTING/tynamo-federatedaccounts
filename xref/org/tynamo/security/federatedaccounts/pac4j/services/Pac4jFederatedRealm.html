49  	public Pac4jFederatedRealm(Logger logger, FederatedAccountService federatedAccountService) {
50  
51  		super(new MemoryConstrainedCacheManager());
52  		this.federatedAccountService = federatedAccountService;
53  		this.logger = logger;
54  
55  		setName(pac4j_.name());
56  		setAuthenticationTokenClass(Pac4jAuthenticationToken.class);
57  	}
58  
59  	@Override
60  	protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authenticationToken)
61  		throws AuthenticationException {
62  
63  		Pac4jAuthenticationToken token = (Pac4jAuthenticationToken) authenticationToken;
64  		UserProfile profile = (UserProfile) token.getPrincipal();
65  		String realmName = pac4j_.name() + profile.getClass().getSimpleName().toLowerCase();
66  		if (realmName.endsWith("profile")) realmName = realmName.substring(0, realmName.lastIndexOf("profile"));
67  		/*
68  		 * OAuthRequest request = new OAuthRequest(Verb.GET, "https://www.google.com/m8/feeds/contacts/default/full");
69  		 * service.signRequest(token.getToken(), request); request.send().getBody();
70  		 */
71  
72  		// Null username is invalid, throw an exception if so - indicates that user hasn't granted the right
73  		// permissions (and/or we haven't asked for it)
74  		/*
75  		 * if (facebookUser == null) throw new AccountException("Null Facebook user is not allowed by this realm.");
76  		 * 
77  		 * String principalValue = null; switch (principalProperty) { case id: principalValue = facebookUser.getId(); break; case email:
78  		 * principalValue = facebookUser.getEmail(); break; case name: principalValue = facebookUser.getName(); break; }
79  		 */
80  
81  		return federatedAccountService.federate(realmName, profile.getId(), authenticationToken, profile);
82  		// SimplePrincipalCollection principalCollection = new SimplePrincipalCollection(authenticationToken.getPrincipal(),
83  		// pac4j.name());
84  		// principalCollection.add(authenticationToken, pac4j.name());
85  		// return new SimpleAuthenticationInfo(principalCollection, authenticationToken.getCredentials());
86  
87  	}
