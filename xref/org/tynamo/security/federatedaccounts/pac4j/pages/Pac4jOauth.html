28  public class Pac4jOauth extends AbstractOauthPage {
29  	@Inject
30  	@Symbol(FederatedAccountSymbols.HTTPCLIENT_ON_GAE)
31  	private boolean httpClientOnGae;
32  
33  	@Inject
34  	private Logger logger;
35  
36  	@Inject
37  	private Request request;
38  
39  	@Inject
40  	private HttpServletRequest httpRequest;
41  
42  	@Inject
43  	private HttpServletResponse httpResponse;
44  
45  	@Inject
46  	private AlertManager alertManager;
47  
48  	// private String returnUri;
49  
50  	@Inject
51  	private Cookies cookies;
52  
53  	@Inject
54  	private Pac4jOauthClientLocator oauthClientLocator;
55  
56  	private String clientName;
57  
58  	@Override
59  	protected Object onOauthActivate(EventContext eventContext) throws Exception {
60  		clientName = eventContext.get(String.class, 1);
61  		BaseOAuthClient<?> client = oauthClientLocator.getClient(clientName);
62  		client.setReadTimeout(20000);
63  		client.setConnectTimeout(20000);
64  		if (eventContext.getCount() > 3) {
65  			String action = eventContext.get(String.class, 2);
66  			// pass along this redirectUrl
67  			setReturnUri(eventContext.get(String.class, 3));
68  			client.setCallbackUrl(getOauthRedirectLink(getWindowMode(), clientName));
69  			if ("request_token".equals(action)) {
70  				String providerOauthUrl = client.getRedirectionUrl(new J2EContext(httpRequest, httpResponse), true);
71  				// fix an issue with pac4j dropboxClient and "direct redirection"
72  				if (client instanceof DropBoxClient)
73  					if (!providerOauthUrl.contains("&oauth_callback"))
74  						providerOauthUrl += "&oauth_callback=" + client.getCallbackUrl();
75  				// Because Google requires the callback uri to match *exactly* (including query parameters) with the configuration,
76  				// use a cookie to store the destination uri after successful Oauth authentication
77  				cookies.writeCookieValue("oauth_returnpage", getReturnUri(), -1);
78  				return new URL(providerOauthUrl);
79  			}
80  		}
81  		// FIXME should we catch and handle the event context errors? Probably
82  
83  		// capture the redirect url again
84  		String returnPageUri = cookies.readCookieValue("oauth_returnpage");
85  		if (returnPageUri != null) {
86  			setReturnUri(returnPageUri);
87  			cookies.removeCookieValue("oauth_returnpage");
88  		}
89  
90  		// we'll get a technical exception if callback url is not set even though it shouldn't be needed anymore
91  		client.setCallbackUrl(getOauthRedirectLink(getWindowMode(), clientName));
92  
93  		OAuthCredentials credentials = client.getCredentials(new J2EContext(httpRequest, httpResponse));
94  
95  		UserProfile userProfile = client.getUserProfile(credentials);
96  		Pac4jAuthenticationToken accessToken = new Pac4jAuthenticationToken(userProfile);
97  
98  		// TODO just use the default rememberMe for now. We could later add support for providing rememberMe in context
99  		accessToken.setRememberMe(isRememberMe());
100 
101 		try {
102 			SecurityUtils.getSubject().login(accessToken);
103 			alertManager.success("User successfully authenticated");
104 			setOauthAuthenticated(true);
105 		} catch (AuthenticationException e) {
106 			logger.error(
107 				"Using access token " + accessToken
108 					+ String.format("\nCould not sign in a %s federated user because of: ", credentials.getClientName()), e);
109 			// FIXME Deal with other account exception types like expired and
110 			// locked
111 			alertManager.error(String.format("A %s federated user cannot be signed in, report this to support.\n %s",
112 				credentials.getClientName(), e.getMessage()));
113 		}
114 		client = null;
115 		return null;
116 	}
117 
118 	public String getProviderPrefix() {
119 		return FederatedAccountType.pac4j_ + clientName;
120 	}
