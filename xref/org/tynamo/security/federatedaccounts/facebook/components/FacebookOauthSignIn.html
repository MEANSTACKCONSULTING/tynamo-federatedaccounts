17  @Import(stylesheet = "fb-button.css")
18  public class FacebookOauthSignIn extends AbstractOauthSignIn {
19  
20  	@Inject
21  	@Symbol(FacebookRealm.FACEBOOK_PERMISSIONS)
22  	@Property
23  	private String facebookPermissions;
24  
25  	@Parameter(defaultPrefix = BindingConstants.LITERAL, value = "800")
26  	@Property
27  	private Integer width;
28  
29  	@Parameter(defaultPrefix = BindingConstants.LITERAL, value = "400")
30  	@Property
31  	private Integer height;
32  
33  	@Parameter(defaultPrefix = BindingConstants.LITERAL, value = "blank")
34  	private WindowMode windowMode;
35  
36  	@Inject
37  	private AssetSource assetSource;
38  
39  	@Inject
40  	private ComponentResources componentResources;
41  
42  	@Inject
43  	private PageRenderLinkSource pageRenderLinkSource;
44  
45  	public boolean isWindowMode(String mode) {
46  		if (mode == null) throw new IllegalArgumentException("Window mode argument cannot be null");
47  		return windowMode.equals(WindowMode.valueOf(mode));
48  	}
49  
50  	public String getOauthAuthorizationLink() {
51  		StringBuilder sb = new StringBuilder();
52  		sb.append("https://graph.facebook.com/oauth/authorize?client_id=");
53  		sb.append(getOauthClientId());
54  		sb.append("&redirect_uri=");
55  		sb.append(getOauthRedirectLink());
56  		sb.append("&display=popup");
57  		sb.append("&scope=");
58  		sb.append(facebookPermissions);
59  
60  		return sb.toString();
61  	}
62  
63  	public String getOauthRedirectLink() {
64  		if ("".equals(getReturnPageName())) return getOauthRedirectLink(windowMode);
65  		if ("^".equals(getReturnPageName()))
66  			return getOauthRedirectLink(windowMode,
67  				pageRenderLinkSource.createPageRenderLink(componentResources.getPage().getClass()).toAbsoluteURI());
68  		return getOauthRedirectLink(windowMode, pageRenderLinkSource.createPageRenderLink(getReturnPageName())
69  			.toAbsoluteURI());
70  	}
71  
72  	@Override
73  	protected Class<?> getOauthPageClass() {
74  		return FacebookOauth.class;
75  	}
