30  public class FacebookOauth extends AbstractOauthPage {
31  	@Inject
32  	private Logger logger;
33  
34  	@Inject
35  	private Request request;
36  
37  	@Inject
38  	private AlertManager alertManager;
39  
40  	@Inject
41  	private PageRenderLinkSource linkSource;

43  	protected Object onOauthActivate(EventContext eventContext) throws MalformedURLException {
44  		String code = request.getParameter("code");
45  
46  		if (eventContext.getCount() > 1) setReturnUri(eventContext.get(String.class, 1));
47  
48  		if (code == null) {
49  			alertManager.error("No Oauth authentication code provided");
50  			return null;
51  		}
52  
53  		List<NameValuePair> qparams = new ArrayList<NameValuePair>();
54  		qparams.add(new BasicNameValuePair("client_id", getOauthClientId()));
55  		qparams.add(new BasicNameValuePair("redirect_uri", getOauthRedirectLink(getWindowMode(), getReturnUri())));
56  		qparams.add(new BasicNameValuePair("client_secret", getOauthClientSecret()));
57  		qparams.add(new BasicNameValuePair("code", code));
58  		HttpGet get = null;
59  		String accessToken = "";
60  		long expires = 0L;
61  		try {
62  			URI uri = URIUtils.createURI("https", "graph.facebook.com", -1, "/oauth/access_token",
63  				URLEncodedUtils.format(qparams, "UTF-8"), null);
64  			get = new HttpGet(uri);
65  
66  			// HttpGet get = new HttpGet(accessTokenUri);
67  			// NameValuePair[] queryString = new NameValuePair[4];
68  			// queryString[0] = new NameValuePair("client_id", clientId);
69  			// queryString[1] = new NameValuePair("redirect_uri", getOauthRedirectLink());
70  			// queryString[2] = new NameValuePair("client_secret", clientSecret);
71  			// queryString[3] = new NameValuePair("code", code);
72  			// get.setQueryString(queryString);
73  			// HttpClient httpClient = new DefaultHttpClient();
74  			HttpClient httpClient = new DefaultHttpClient();
75  
76  			HttpResponse response = httpClient.execute(get);
77  			int status = response.getStatusLine().getStatusCode();
78  			if (HttpStatus.SC_OK != status) {
79  				logger.error("Facebook access_token request returned status code " + status);
80  				alertManager.error("Facebook access_token request failed with status code: " + status);
81  				return null;
82  			}
83  			HttpEntity entity = response.getEntity();
84  			if (entity != null) {
85  				long len = entity.getContentLength();
86  				if (len != -1 && len < 2048) accessToken = EntityUtils.toString(entity);
87  			}
88  		} catch (Exception e) {
89  			logger.error("Facebook access_token request failed because of:", e);
90  			alertManager.error("Facebook access_token request failed with message: " + e.getMessage());
91  			return null;
92  		} finally {
93  			if (get != null) get.abort();
94  		}
95  
96  		try {
97  			if (!accessToken.startsWith("access_token")) throw new IllegalArgumentException();
98  			// access_token is of form:
99  			// access_token=119507274749030|2.1AptZFp9__qW3k2PuG4bVA__.3600.1274914800-539598633|9aTyryhVl8vnn3ulLy2w6Txo92E.&expires=4059
100 			accessToken = accessToken.substring(accessToken.indexOf("=") + 1);
101 			expires = Long.valueOf(accessToken.substring(accessToken.lastIndexOf("=") + 1));
102 			accessToken = accessToken.substring(0, accessToken.indexOf("&expires"));
103 		} catch (Exception e) {
104 			logger.error("access_token wasn't of right format");
105 			alertManager.error("Facebook access_token wasn't of right format");
106 			return null;
107 		}
108 
109 		FacebookAccessToken fbAccessToken = new FacebookAccessToken(accessToken, expires);
110 		fbAccessToken.setRememberMe(isRememberMe());
111 		try {
112 			SecurityUtils.getSubject().login(fbAccessToken);
113 			alertManager.success("User successfully authenticated");
114 			setOauthAuthenticated(true);
115 		} catch (AuthenticationException e) {
116 			logger.error("Using access token " + accessToken + "\nCould not sign in a Facebook federated user because of: ",
117 				e);
118 			// FIXME Deal with other account exception types like expired and
119 			// locked
120 			alertManager.error("A Facebook federated user cannot be signed in, report this to support.\n "
121 				+ e.getMessage());
122 			return null;
123 		}
124 		return null;
125 	}
126 }
