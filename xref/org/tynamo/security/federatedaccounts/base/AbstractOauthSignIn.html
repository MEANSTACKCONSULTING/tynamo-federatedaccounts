1   package org.tynamo.security.federatedaccounts.base;
2   
3   import org.apache.tapestry5.BindingConstants;
4   import org.apache.tapestry5.ComponentResources;
5   import org.apache.tapestry5.annotations.Import;
6   import org.apache.tapestry5.annotations.Parameter;
7   import org.apache.tapestry5.annotations.Property;
8   import org.apache.tapestry5.ioc.annotations.Inject;
9   import org.apache.tapestry5.ioc.annotations.Symbol;
10  import org.apache.tapestry5.services.BaseURLSource;
11  import org.apache.tapestry5.services.PageRenderLinkSource;
12  import org.apache.tapestry5.services.Request;
13  import org.tynamo.security.federatedaccounts.FederatedAccountSymbols;
14  
15  @Import(library = "OauthSignIn.js")
16  public abstract class AbstractOauthSignIn extends OauthComponentBase {
17  	// Use ^ for current page the component is contained in, empty for configured success url
18  	@Parameter(defaultPrefix = BindingConstants.LITERAL, value = "prop:defaultReturnPage")
19  	String returnPageName;
20  
21  	public String getReturnPageName() {
22  		return returnPageName;
23  	}
24  
25  	@Inject
26  	@Symbol(FederatedAccountSymbols.DEFAULT_RETURNPAGE)
27  	@Property(write = false)
28  	String defaultReturnPage;
29  
30  	@Inject
31  	PageRenderLinkSource pageRenderLinkSource;
32  
33  	@Inject
34  	ComponentResources componentResources;
35  
36  	@Inject
37  	BaseURLSource baseURLSource;
38  
39  	@Inject
40  	Request request;
41  
42  	public String getReturnPageUri() {
43  		if ("^".equals(returnPageName))
44  			return pageRenderLinkSource.createPageRenderLink(componentResources.getPage().getClass()).toAbsoluteURI();
45  
46  		// Empty returnpageName denotes a base url. Default for default return page may be set to a non empty value.
47  		// If default is empty and not locally overridden, return the baseurl
48  		return "".equals(returnPageName) && "".equals(defaultReturnPage) ? baseURLSource.getBaseURL(request.isSecure())
49  			: pageRenderLinkSource.createPageRenderLink(returnPageName).toAbsoluteURI();
50  
51  	}
52  }
