3   import javax.servlet.http.Cookie;
4   import javax.servlet.http.HttpServletRequest;
5   
6   import org.apache.shiro.authc.AuthenticationException;
7   import org.apache.shiro.subject.PrincipalCollection;
8   import org.apache.shiro.subject.Subject;
9   import org.apache.tapestry5.plastic.MethodAdvice;
10  import org.apache.tapestry5.plastic.MethodInvocation;
11  import org.slf4j.Logger;
12  import org.tynamo.security.rollingtokens.services.RollingToken;
13  
14  public class RollingTokenAutoLoginAdvice implements MethodAdvice {
15  	private final Logger logger;
16  	private final HttpServletRequest request;
17  	private String realmName;
18  	private Class<?> principalType;
19  
20  	public RollingTokenAutoLoginAdvice(Logger logger, HttpServletRequest request, String realmName, Class<?> principalType)
21  		throws ClassNotFoundException {
22  		this.logger = logger;
23  		this.request = request;
24  		this.realmName = realmName;
25  		this.principalType = principalType;
26  	}
27  
28  	@Override
29  	public void advise(MethodInvocation invocation) {
30  		invocation.proceed();
31  		// SubjectContext context = (SubjectContext) invocation.getParameter(0);
32  		// if (!(context instanceof WebSubjectContext)) { return; }
33  		// WebSubjectContext wsc = (WebSubjectContext) context;
34  		// PrincipalCollection principals = wsc.resolvePrincipals();
35  		// boolean authenticated = wsc.resolveAuthenticated();
36  		Subject subject = (Subject) invocation.getReturnValue();
37  		PrincipalCollection principals = subject.getPrincipals();
38  		if (principals == null || subject.isAuthenticated()) return;
39  
40  		String rollingTokenValue = null;
41  		try {
42  			// one issue is that underlying request shadow may at times be null
43  			// A bigger problem is that when security is decorating the request, it needs a subject
44  			// but we are only building it here. So we can't use Request shadow
45  			Cookie[] cookies = request.getCookies();
46  			// System.out.println("request " + request + " " + request.getCookies());
47  			for (Cookie cookie : cookies)
48  				if (RollingToken.TOKEN_NAME.equals(cookie.getName())) {
49  					rollingTokenValue = cookie.getValue();
50  					break;
51  				}
52  		} catch (RuntimeException e) {
53  			// ignore
54  		}
55  		if (rollingTokenValue == null) return;
56  
57  		Object principal = RollingToken.getConfiguredPrincipal(realmName, principalType, principals);
58  
59  		RollingToken token = new RollingToken(principal, rollingTokenValue, request.getRemoteAddr());
60  		try {
61  			subject.login(token);
62  			logger.info("Rolling token authentication using token " + rollingTokenValue
63  				+ " succeeded for user identified by " + principal);
64  		} catch (AuthenticationException e) {
65  			logger.info("Rolling token authentication using token " + rollingTokenValue + " failed for user identified by "
66  				+ principal);
67  		}
68  
69  	}
70  
71  }
