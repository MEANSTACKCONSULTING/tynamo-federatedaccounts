32  public class UserRealm extends AuthorizingRealm {
33  	protected final EntityManager em;
34  
35  	public UserRealm(EntityManager em) {
36  		super(new MemoryConstrainedCacheManager());
37  		this.em = em;
38  		setName("localaccounts");
39  		setAuthenticationTokenClass(UsernamePasswordToken.class);
40  		setCredentialsMatcher(new HashedCredentialsMatcher(Sha1Hash.ALGORITHM_NAME));
41  	}
42  
43  	@Override
44  	protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
45  		if (principals == null) throw new AuthorizationException("PrincipalCollection was null, which should not happen");
46  
47  		if (principals.isEmpty()) return null;
48  
49  		if (principals.fromRealm(getName()).size() <= 0) return null;
50  
51  		String username = (String) principals.fromRealm(getName()).iterator().next();
52  		if (username == null) return null;
53  		User user = findByUsername(username);
54  		if (user == null) return null;
55  		Set<String> roles = new HashSet<String>(user.getRoles().size());
56  		for (Role role : user.getRoles())
57  			roles.add(role.name());
58  		return new SimpleAuthorizationInfo(roles);
59  	}
60  
61  	private User findByUsername(String username) {
62  		CriteriaBuilder qb = em.getCriteriaBuilder();
63  		CriteriaQuery<User> query = qb.createQuery(User.class);
64  		Root<User> root = query.from(User.class);
65  		List<User> results = em.createQuery(query.where(qb.equal(root.get("username"), username))).getResultList();
66  		return results.size() <= 0 ? null : results.get(0);
67  	}
68  
69  	@Override
70  	protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
71  		UsernamePasswordToken upToken = (UsernamePasswordToken) token;
72  
73  		String username = upToken.getUsername();
74  
75  		// Null username is invalid
76  		if (username == null) { throw new AccountException("Null usernames are not allowed by this realm."); }
77  
78  		User user = findByUsername(username);
79  		if (user.getFacebookUserId() != null) { throw new AccountException("Account [" + username
80  				+ "] is federated with Facebook and cannot be locally authenticated."); }
81  
82  		if (user.isAccountLocked()) { throw new LockedAccountException("Account [" + username + "] is locked."); }
83  		if (user.isCredentialsExpired()) {
84  			String msg = "The credentials for account [" + username + "] are expired";
85  			throw new ExpiredCredentialsException(msg);
86  		}
87  		return new SimpleAuthenticationInfo(username, user.getEncodedPassword(), new SimpleByteSource(user.getPasswordSalt()), getName());
88  	}

90  }
