42  	public static void contributeApplicationDefaults(MappedConfiguration<String, String> configuration) {
43  		// Contributions to ApplicationDefaults will override any contributions to
44  		// FactoryDefaults (with the same key). Here we're restricting the supported
45  		// locales to just "en" (English). As you add localised message catalogs and other assets,
46  		// you can extend this list of locales (it's a comma separated series of locale names;
47  		// the first locale name is the default when there's no reasonable match).
48  
49  		configuration.add(SymbolConstants.SUPPORTED_LOCALES, "en");
50  
51  		// The factory default is true but during the early stages of an application
52  		// overriding to false is a good idea. In addition, this is often overridden
53  		// on the command line as -Dtapestry.production-mode=false
54  		configuration.add(SymbolConstants.PRODUCTION_MODE, "false");
55  
56  		// The application version number is incorprated into URLs for some
57  		// assets. Web browsers will cache assets because of the far future expires
58  		// header. If existing assets are changed, the version number should also
59  		// change, to force the browser to download new versions.
60  		configuration.add(SymbolConstants.APPLICATION_VERSION, "0.0.1-SNAPSHOT");
61  
62  		configuration.add(FederatedAccountSymbols.LOCALACCOUNT_REALMNAME, "localaccounts");
63  
64  		configuration.add(SecuritySymbols.LOGIN_URL, "/login");
65  		configuration.add(SecuritySymbols.SUCCESS_URL, "/index");
66  
67  //		configuration.add(Pac4jFederatedRealm.YAHOO_CLIENTID, "somekey");
68  //		configuration.add(Pac4jFederatedRealm.YAHOO_CLIENTSECRET, "just to test yahoo login icon");
69  	}
