042        public static void contributeApplicationDefaults(MappedConfiguration<String, String> configuration) {
043                // Contributions to ApplicationDefaults will override any contributions to
044                // FactoryDefaults (with the same key). Here we're restricting the supported
045                // locales to just "en" (English). As you add localised message catalogs and other assets,
046                // you can extend this list of locales (it's a comma separated series of locale names;
047                // the first locale name is the default when there's no reasonable match).
048
049                configuration.add(SymbolConstants.SUPPORTED_LOCALES, "en");
050
051                // The factory default is true but during the early stages of an application
052                // overriding to false is a good idea. In addition, this is often overridden
053                // on the command line as -Dtapestry.production-mode=false
054                configuration.add(SymbolConstants.PRODUCTION_MODE, "false");
055
056                // The application version number is incorprated into URLs for some
057                // assets. Web browsers will cache assets because of the far future expires
058                // header. If existing assets are changed, the version number should also
059                // change, to force the browser to download new versions.
060                configuration.add(SymbolConstants.APPLICATION_VERSION, "0.0.1-SNAPSHOT");
061
062                configuration.add(FederatedAccountSymbols.LOCALACCOUNT_REALMNAME, "localaccounts");
063
064                configuration.add(SecuritySymbols.LOGIN_URL, "/login");
065                configuration.add(SecuritySymbols.SUCCESS_URL, "/index");
066
067//              configuration.add(Pac4jFederatedRealm.YAHOO_CLIENTID, "somekey");
068//              configuration.add(Pac4jFederatedRealm.YAHOO_CLIENTSECRET, "just to test yahoo login icon");
069        }

071        public static void contributeSecurityConfiguration(Configuration<SecurityFilterChain> configuration,
072                        SecurityFilterChainFactory factory) {
073                configuration.add(factory.createChain("/assets/**").add(factory.anon()).build());
074                configuration.add(factory.createChain("/login.loginform.tynamologinform").add(factory.anon()).build());
075                configuration.add(factory.createChain("/federated/**").add(factory.anon()).build());
076                configuration.add(factory.createChain("/modules/**").add(factory.anon()).build());
077                configuration.add(factory.createChain("/modules.gz/**").add(factory.anon()).build());
078                configuration.add(factory.createChain("/**").add(factory.authc()).build());
079        }

086        public static void contributeSeedEntity(OrderedConfiguration<Object> configuration) {
087                User localUser = new User();
088                localUser.setUsername("user");
089                localUser.setFirstName("Local");
090                localUser.setLastName("User");
091                localUser.setPassword("user");
092                configuration.add("localuser", localUser);
093                User fakeFederatedUser = new User();
094                fakeFederatedUser.setUsername("fbuser");
095                fakeFederatedUser.setFirstName("Facebook");
096                fakeFederatedUser.setLastName("User");
097                fakeFederatedUser.setFacebookUserId(0L);
098                configuration.add("fakeuser", fakeFederatedUser);
099        }
