001package org.tynamo.security.federatedaccounts.testapp.pages;
002
003import org.apache.tapestry5.services.ExceptionReporter;
004
005public class Login implements ExceptionReporter {
006
007        private Throwable exception;
008
009        @Override
010        public void reportException(Throwable exception) {
011                this.exception = exception;
012        }
013
014        public Throwable getException() {
015                return exception;
016        }
017
018        public String getMessage() {
019                if (exception != null) {
020                        return exception.getMessage() + " Try login.";
021                } else {
022                        return "";
023                }
024        }
025
026}
