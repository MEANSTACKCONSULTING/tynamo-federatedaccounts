001package org.tynamo.security.federatedaccounts.base;
002
003import static org.mockito.Matchers.any;
004import static org.mockito.Mockito.mock;
005import static org.mockito.Mockito.when;
006import static org.testng.AssertJUnit.assertEquals;
007
008import org.apache.tapestry5.ComponentResources;
009import org.apache.tapestry5.Link;
010import org.apache.tapestry5.runtime.Component;
011import org.apache.tapestry5.services.BaseURLSource;
012import org.apache.tapestry5.services.PageRenderLinkSource;
013import org.apache.tapestry5.services.Request;
014import org.testng.annotations.BeforeMethod;
015import org.testng.annotations.Test;
016
017public class AbstractOauthSignInUnitTest {
018        Link link = mock(Link.class);
019        AbstractOauthSignIn oauthSignIn = new AbstractOauthSignIn() {
020
021                @Override
022                protected Class getOauthPageClass() {
023                        return getClass();
024                }
025        };
026
027        @BeforeMethod
028        public void setUp() {
029                oauthSignIn.baseURLSource = mock(BaseURLSource.class);
030                oauthSignIn.request = mock(Request.class);
031                oauthSignIn.pageRenderLinkSource = mock(PageRenderLinkSource.class);
032                oauthSignIn.baseURLSource = mock(BaseURLSource.class);
033                oauthSignIn.componentResources = mock(ComponentResources.class);
034
035                when(oauthSignIn.pageRenderLinkSource.createPageRenderLink((Class) any())).thenReturn(link);
036                when(oauthSignIn.pageRenderLinkSource.createPageRenderLink((String) any())).thenReturn(link);
037                Component page = mock(Component.class);
038                when(oauthSignIn.componentResources.getPage()).thenReturn(page);
039        }
040
041        @Test
042        public void getReturnPageUriWithEmptyReturnPage() {
043                oauthSignIn.defaultReturnPage = "";
044                oauthSignIn.returnPageName = "";
045
046                when(oauthSignIn.baseURLSource.getBaseURL(false)).thenReturn("http://givemethebaseurl.com/");
047
048                assertEquals("http://givemethebaseurl.com/", oauthSignIn.getReturnPageUri());
049        }
050
051        @Test
052        public void getReturnPageUriWithReturnPageSpecified() {
053                oauthSignIn.returnPageName = "home";
054
055                when(link.toAbsoluteURI()).thenReturn("http://foo.com/home");
056                assertEquals("http://foo.com/home", oauthSignIn.getReturnPageUri());
057                // we cannot test for defaultReturnPage unless we processed the Tapestry annotation one way or another
058        }
059
060        @Test
061        public void getReturnPageUriForContainingPage() {
062                oauthSignIn.returnPageName = "^";
063
064                when(link.toAbsoluteURI()).thenReturn("http://foo.com/some");
065                assertEquals("http://foo.com/some", oauthSignIn.getReturnPageUri());
066        }
067
068}
