070        protected void assertLoginPage(HtmlPage page) {
071                assertNotNull(page.getElementById("tynamoLogin"), "Page doesn't contain login field. Not a login page.");
072                assertEquals("password", getAttribute("tynamoPassword", "type"),
073                        "Page doesn't contain password field. Not a login page.");
074                assertEquals("checkbox", getAttribute("tynamoRememberMe", "type"),
075                        "Page doesn't contain rememberMe field. Not a login page.");
076
077                assertNotNull(page.getElementById("tynamoEnter"),
078                        "Page doesn't contain login form submit button. Not a login page.");
079        }
080
081        // -----------------------
082
083        private String getAttribute(String id, String attr) {
084                return page.getElementById(id).getAttribute(attr);
085        }
086
087        protected void assertUnauthorizedPage() {
088                assertEquals(getTitle(), "Unauthorized", "Not Unauthorized page");
089        }
090
091        protected void assertUnauthorizedPage401() {
092                assertEquals(getTitle(), "Error 401 UNAUTHORIZED", "Not Unauthorized page");
093        }
094
095        protected void openPage(String url) throws Exception {
096                page = webClient.getPage(BASEURI + url);
097        }
098
099        protected void openBase() throws Exception {
100                openPage("");
101        }
102
103        protected void clickOnBasePage(String url) throws Exception {
104                openBase();
105                page = page.getHtmlElementById(url).click();
106        }
107
108        protected void assertAuthenticated() {
109                assertEquals(getText("status"), STATUS_AUTH);
110        }
111
112        protected void assertNotAuthenticated() {
113                assertEquals(getText("status"), STATUS_NOT_AUTH);
114        }
115
116        protected void signIn(String username, String password) throws IOException {
117                type("tynamoLogin", username);
118                type("tynamoPassword", password);
119                page = click("tynamoEnter");
120        }
121
122        private void type(String id, String value) {
123                page.getForms().get(0).<HtmlInput> getInputByName(id).setValueAttribute(value);
124        }
125
126        private HtmlPage click(String id) throws IOException {
127                return clickButton(page, id);
128        }
129
130        private String getText(String id) {
131                return page.getElementById(id).asText();
132        }
133
134        private void assertText(String id, String text) {
135                assertEquals(page.getElementById(id).asText(), text);
136        }
137
138        private boolean isElementPresent(String id) {
139                return page.getElementById(id) != null;
140        }
141
142        private String getTitle() {
143                return page.getTitleText();
144        }
145
146        private String getLocation() {
147                return page.getWebResponse().getWebRequest().getUrl().toString();
148        }
